50 static BOOL SendWebAnalyticsRequest(char *url)
51 {
52  URLMON_PROCEDURE pURLDownloadToCacheFile;
53  HMODULE hUrlmonDLL = NULL;
54  HRESULT result;
55  char path[MAX_PATH + 1] = {0};
56 
57  if(url == NULL){
58  WgxDbgPrint("SendWebAnalyticsRequest: URL = NULL\n");
59  goto fail;
60  }
61 
62  WgxDbgPrint("SendWebAnalyticsRequest: URL = %s\n",url);
63 
64  /* load urlmon.dll library */
65  hUrlmonDLL = LoadLibrary("urlmon.dll");
66  if(hUrlmonDLL == NULL){
67  WgxDbgPrintLastError("SendWebAnalyticsRequest: LoadLibrary(urlmon.dll) failed");
68  goto fail;
69  }
70 
71  /* get an address of procedure downloading a file */
72  pURLDownloadToCacheFile = (URLMON_PROCEDURE)GetProcAddress(hUrlmonDLL,"URLDownloadToCacheFileA");
73  if(pURLDownloadToCacheFile == NULL){
74  WgxDbgPrintLastError("SendWebAnalyticsRequest: URLDownloadToCacheFile not found in urlmon.dll");
75  goto fail;
76  }
77 
78  /* download a file */
79  result = pURLDownloadToCacheFile(NULL,url,path,MAX_PATH,0,NULL);
80  path[MAX_PATH] = 0;
81  if(result != S_OK){
82  if(result == E_OUTOFMEMORY)
83  WgxDbgPrint("SendWebAnalyticsRequest: not enough memory\n");
84  else
85  WgxDbgPrint("SendWebAnalyticsRequest: URLDownloadToCacheFile failed\n");
86  goto fail;
87  }
88 
89  /* remove cached data, otherwise it may not be loaded next time */
90  if (path[0] != 0) (void)remove(path);
91 
92  /* free previously allocated url */
93  free(url);
94  return TRUE;
95 
96 fail:
97  free(url);
98  return FALSE;
99 }
