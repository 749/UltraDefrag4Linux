104  return FALSE;
105  }
106 
107  /* stop collector during initialization */
108  lua_gc(L, LUA_GCSTOP, 0);
109  luaL_openlibs(L);
110  lua_gc(L, LUA_GCRESTART, 0);
111 
112  status = luaL_dofile(L,path);
113  if(status != 0){
114  WgxDbgPrint("WgxGetOptions: cannot interprete %s",path);
115  if(!lua_isnil(L, -1)){
116  msg = lua_tostring(L, -1);
117  if(msg == NULL) msg = "(error object is not a string)";
118  WgxDbgPrint("WgxGetOptions: %s",msg);
119  lua_pop(L, 1);
120  }
121  lua_close(L);
122  return FALSE;
123  }
124 
125  /* search for variables */
126  for(i = 0; table[i].name; i++){
127  lua_getglobal(L, table[i].name);
128  if(!lua_isnil(L, lua_gettop(L))){
129  if(table[i].type == WGX_CFG_INT){
130  *((int *)table[i].value) = (int)lua_tointeger(L, lua_gettop(L));
131  } else if(table[i].type == WGX_CFG_STRING){
132  s = (char *)lua_tostring(L, lua_gettop(L));
133  if(s != NULL){
134  strncpy((char *)table[i].value,s,table[i].value_length);
135  ((char *)table[i].value)[table[i].value_length - 1] = 0;
136  } else {
137  strcpy((char *)table[i].value,"");
138  }
139  }
140  }
141  /* important: remove received variable from stack */
142  lua_pop(L, 1);
143  }
