73 BOOL WgxCreateFont(char *wgx_font_path,PWGX_FONT pFont)
74 {
75  lua_State *L;
76  int status;
77  char *string;
78  LOGFONT lf;
79 
80  if(pFont == NULL)
81  return FALSE;
82 
83  /* don't reset lf, it may contain default application's font */
84  pFont->hFont = NULL;
85 
86  if(wgx_font_path == NULL)
87  goto use_default_font;
88 
89  L = lua_open(); /* create state */
90  if(L == NULL){
91  WgxDbgPrint("WgxCreateFont: cannot initialize Lua library\n");
92  goto use_default_font;
93  }
94 
95  lua_gc(L, LUA_GCSTOP, 0); /* stop collector during initialization */
96  luaL_openlibs(L); /* open libraries */
97  lua_gc(L, LUA_GCRESTART, 0);
98 
99  status = luaL_dofile(L,wgx_font_path);
100  if(!status){ /* successful */
101  lf.lfHeight = getint(L,"height");
102  lf.lfWidth = getint(L,"width");
103  lf.lfEscapement = getint(L,"escapement");
104  lf.lfOrientation = getint(L,"orientation");
105  lf.lfWeight = getint(L,"weight");
106  lf.lfItalic = (BYTE)getint(L,"italic");
107  lf.lfUnderline = (BYTE)getint(L,"underline");
108  lf.lfStrikeOut = (BYTE)getint(L,"strikeout");
109  lf.lfCharSet = (BYTE)getint(L,"charset");
110  lf.lfOutPrecision = (BYTE)getint(L,"outprecision");
111  lf.lfClipPrecision = (BYTE)getint(L,"clipprecision");
112  lf.lfQuality = (BYTE)getint(L,"quality");
113  lf.lfPitchAndFamily = (BYTE)getint(L,"pitchandfamily");
114  lua_getglobal(L, "facename");
115  string = (char *)lua_tostring(L, lua_gettop(L));
116  if(string){
117  (void)strncpy(lf.lfFaceName,string,LF_FACESIZE);
118  lf.lfFaceName[LF_FACESIZE - 1] = 0;
119  } else {
120  lf.lfFaceName[0] = 0;
121  }
122  lua_pop(L, 1);
123  lua_close(L);
124  } else {
125  WgxDbgPrint("WgxCreateFont: cannot interprete %s\n",wgx_font_path);
126  lua_close(L);
127  goto use_default_font;
128  }
129 
130  pFont->hFont = CreateFontIndirect(&lf);
131  if(pFont->hFont == NULL){
132  WgxDbgPrintLastError("WgxCreateFont: CreateFontIndirect for custom font failed");
133 use_default_font:
134  /* try to use default font passed through pFont */
135  pFont->hFont = CreateFontIndirect(&pFont->lf);
136  if(pFont->hFont == NULL){
137  WgxDbgPrintLastError("WgxCreateFont: CreateFontIndirect for default font failed");
138  return FALSE;
139  }
140  return TRUE;
141  }
142 
143  memcpy(&pFont->lf,&lf,sizeof(LOGFONT));
144  return TRUE;
145 }

152 void WgxSetFont(HWND hWnd, PWGX_FONT pFont)
153 {
154  HWND hChild;
155 
156  if(pFont == NULL)
157  return;
158

160  return;
161 
162  (void)SendMessage(hWnd,WM_SETFONT,(WPARAM)pFont->hFont,MAKELPARAM(TRUE,0));
163  hChild = GetWindow(hWnd,GW_CHILD);
164  while(hChild){
165  (void)SendMessage(hChild,WM_SETFONT,(WPARAM)pFont->hFont,MAKELPARAM(TRUE,0));
166  hChild = GetWindow(hChild,GW_HWNDNEXT);
167  }
168 
169  /* redraw the main window */
170  (void)InvalidateRect(hWnd,NULL,TRUE);
171  (void)UpdateWindow(hWnd);
172 }
