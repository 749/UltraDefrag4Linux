91 BOOL WgxLoadIcon(HINSTANCE hInstance,UINT IconID,UINT size,HICON *phIcon)
92 {
93  LOAD_ICON_METRIC_PROC pLoadIconMetric;
94  HMODULE hLib;
95  HRESULT hr;
96  int is_standard_icon_size = 0;
97  int lims = 0;
98  HICON ScaledIcon;
99 
100  /* validate parameters */
101  if(size == 0 || phIcon == NULL) return FALSE;
102  *phIcon = NULL;
103 
104  /* do we need to load a standard small/large icon? */
105  if(size == GetSystemMetrics(SM_CXSMICON)){
106  is_standard_icon_size = 1, lims = LIM_SMALL;
107  } else if(size == GetSystemMetrics(SM_CXICON)){
108  is_standard_icon_size = 1, lims = LIM_LARGE;
109  }
110 
111  /*
112  * On Windows Vista and more recent Windows
113  * editions an excellent LoadIconMetric routine
114  * exists.
115  */
116  if(is_standard_icon_size){
117  hLib = LoadLibrary("comctl32.dll");
118  if(hLib == NULL){
119  WgxDbgPrintLastError("WgxLoadIcon: cannot load comctl32.dll library");
120  } else {
121  pLoadIconMetric = (LOAD_ICON_METRIC_PROC)GetProcAddress(hLib,"LoadIconMetric");
122  if(pLoadIconMetric == NULL){
123  WgxDbgPrintLastError("WgxLoadIcon: LoadIconMetric procedure not found in comctl32.dll library");
124  } else {
125  hr = pLoadIconMetric(hInstance,MAKEINTRESOURCEW(IconID),lims,phIcon);
126  if(hr != S_OK || *phIcon == NULL){
127  WgxDbgPrint("WgxLoadIcon: LoadIconMetric failed with code 0x%x",(UINT)hr);
128  } else {
129  return TRUE;
130  }
131  }
132  }
133  }
134 
135  /*
136  * LoadIconMetric routine emulation starts here.
137  * Try to load icon of exact size first.
138  */
139  *phIcon = (HICON)LoadImage(hInstance,
140  MAKEINTRESOURCE(IconID),
141  IMAGE_ICON,size,size,
142  LR_DEFAULTCOLOR);
143  if(*phIcon == NULL){
144  WgxDbgPrintLastError("WgxLoadIcon: LoadImage failed (case 1)");
145  } else {
146  return TRUE;
147  }
148 
149  /*
150  * Icon of the specified size
151  * not found, so try to scale
152  * a standard large icon.
153  */
154  *phIcon = LoadIcon(hInstance,MAKEINTRESOURCE(IconID));
155  if(*phIcon == NULL){
156  WgxDbgPrintLastError("WgxLoadIcon: LoadIcon failed");
157  } else {
158  ScaledIcon = (HICON)CopyImage((HANDLE)*phIcon,
159  IMAGE_ICON,size,size,0);
160  if(ScaledIcon == NULL){
161  WgxDbgPrintLastError("WgxLoadIcon: CopyImage failed (case 1)");
162  DestroyIcon(*phIcon);
163  *phIcon = NULL;
164  } else {
165  DestroyIcon(*phIcon);
166  *phIcon = ScaledIcon;
167  return TRUE;
168  }
169  }
170 
171  /*
172  * A standard large icon does not exist,
173  * so try to scale at least something.
174  */
175  *phIcon = (HICON)LoadImage(hInstance,
176  MAKEINTRESOURCE(IconID),
177  IMAGE_ICON,0,0,
178  LR_DEFAULTCOLOR);
179  if(*phIcon == NULL){
180  WgxDbgPrintLastError("WgxLoadIcon: LoadImage failed (case 2)");
181  } else {
182  ScaledIcon = (HICON)CopyImage((HANDLE)*phIcon,
183  IMAGE_ICON,size,size,0);
184  if(ScaledIcon == NULL){
185  WgxDbgPrintLastError("WgxLoadIcon: CopyImage failed (case 2)");
186  DestroyIcon(*phIcon);
187  *phIcon = NULL;
188  return FALSE;
189  } else {
190  DestroyIcon(*phIcon);
191  *phIcon = ScaledIcon;
192  }
193  }
194  return TRUE;
195 }

223 void WgxCheckWindowCoordinates(LPRECT lprc,int min_width,int min_height)
224 {
225  int cx,cy;
226 
227  cx = GetSystemMetrics(SM_CXSCREEN);
228  cy = GetSystemMetrics(SM_CYSCREEN);
229  if(lprc->left < 0) lprc->left = 0; if(lprc->top < 0) lprc->top = 0;
230  if(lprc->left >= (cx - min_width)) lprc->left = cx - min_width;
231  if(lprc->top >= (cy - min_height)) lprc->top = cy - min_height;

239 void WgxCenterWindow(HWND hwnd)
240 {
241  HWND hwndParent;
242  RECT rect, rectP;
243  int width, height;
244  int screenwidth, screenheight;
245  int x, y;
246 
247  //make the window relative to its parent
248  hwndParent = GetParent(hwnd);
249  if(hwndParent == NULL)
250  hwndParent = GetDesktopWindow();
251 
252  GetWindowRect(hwnd, &rect);
253  GetWindowRect(hwndParent, &rectP);
254 
255  width = rect.right - rect.left;
256  height = rect.bottom - rect.top;
257 
258  x = ((rectP.right-rectP.left) - width) / 2 + rectP.left;
259  y = ((rectP.bottom-rectP.top) - height) / 2 + rectP.top;
260 
261  screenwidth = GetSystemMetrics(SM_CXSCREEN);
262  screenheight = GetSystemMetrics(SM_CYSCREEN);
263 
264  //make sure that the dialog box never moves outside of
265  //the screen
266  if(x < 0) x = 0;
267  if(y < 0) y = 0;
268  if(x + width > screenwidth) x = screenwidth - width;
269  if(y + height > screenheight) y = screenheight - height;
270 
271  MoveWindow(hwnd, x, y, width, height, FALSE);
272 }

289 BOOL WgxGetTextDimensions(wchar_t *text,HFONT hFont,HWND hWnd,int *pWidth,int *pHeight)
290 {
291  HDC hdc;
292  HFONT hOldFont;
293  SIZE size;
294  BOOL result;
295 
296  /* validate parameters */
297  if(pWidth == NULL || pHeight == NULL) return FALSE;
298  *pWidth = *pHeight = 0;
299 
300  if(text == NULL)
301  return TRUE;
302 
303  /* get default font if requested */
304  if(hFont == NULL){
305  hFont = (HFONT)SendMessage(hWnd,WM_GETFONT,0,0);
306  if(hFont == NULL){
307  WgxDbgPrintLastError("WgxGetTextDimensions: cannot get default font");
308  return FALSE;
309  }
310  }
311 
312  /* get dimensions of text */
313  hdc = GetDC(hWnd);
314  if(hdc == NULL){
315  WgxDbgPrintLastError("WgxGetTextDimensions: cannot get device context of the window");
316  return FALSE;
317  }
318  hOldFont = SelectObject(hdc,hFont);
319  result = GetTextExtentPoint32W(hdc,text,wcslen(text),&size);
320  if(result == FALSE){
321  WgxDbgPrintLastError("WgxGetTextDimensions: cannot get text dimensions");
322  } else {
323  *pWidth = size.cx;
324  *pHeight = size.cy;
325  }
326  SelectObject(hdc,hOldFont);
327  ReleaseDC(hWnd,hdc);
328  return result;
329 }
