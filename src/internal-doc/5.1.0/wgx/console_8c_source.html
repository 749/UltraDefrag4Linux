78 void WgxPrintUnicodeString(wchar_t *string,FILE *f)
79 {
80  OSVERSIONINFO osvi;
81  HANDLE hOut;
82  DWORD file_type;
83  DWORD mode;
84  HMODULE hKernel32Dll = NULL;
85  GET_CURRENT_CONSOLE_FONT_EX_PROC pGetCurrentConsoleFontEx = NULL;
86  WIN_CONSOLE_FONT_INFO_EX cfie;
87  UINT old_code_page, new_code_page = CP_OEMCP;
88  char *cnv_string = NULL;
89  int length;
90 
91  if(string == NULL) return;
92 
93  old_code_page = GetConsoleOutputCP();
94 
95  memset(&osvi,0,sizeof(osvi));
96  osvi.dwOSVersionInfoSize = sizeof(osvi);
97  GetVersionEx(&osvi);
98  if(osvi.dwMajorVersion <= 4){
99  /* Windows NT 4.0 supports OEM encoding only */
100  goto convert;
101  }
102 
103  /* output redirected to a file can be converted to UTF-8 */
104  hOut = GetStdHandle(f == stdout ? STD_OUTPUT_HANDLE : STD_ERROR_HANDLE);
105  if(hOut == INVALID_HANDLE_VALUE){
106  handle_api_failure("WgxPrintUnicodeString","GetStdHandle");
107  } else {
108  file_type = GetFileType(hOut);
109  if(file_type == FILE_TYPE_UNKNOWN && GetLastError() != NO_ERROR){
110  handle_api_failure("WgxPrintUnicodeString","GetFileType");
111  } else {
112  file_type &= ~(FILE_TYPE_REMOTE);
113  if(file_type == FILE_TYPE_CHAR){
114  if(!GetConsoleMode(hOut,&mode)){
115  if(GetLastError() == ERROR_INVALID_HANDLE){
116  new_code_page = CP_UTF8;
117  goto convert;
118  } else {
119  handle_api_failure("WgxPrintUnicodeString","GetConsoleMode");
120  }
121  }
122  } else {
123  new_code_page = CP_UTF8;
124  goto convert;
125  }
126  }
127  }
128 
129  /* if the console uses a TrueType font, UTF-8 can be used too */
130  hKernel32Dll = LoadLibrary("kernel32.dll");
131  if(hKernel32Dll == NULL){
132  handle_api_failure("WgxPrintUnicodeString","LoadLibrary");
133  } else {
134  pGetCurrentConsoleFontEx = (GET_CURRENT_CONSOLE_FONT_EX_PROC)GetProcAddress(hKernel32Dll,"GetCurrentConsoleFontEx");
135  if(pGetCurrentConsoleFontEx){
136  memset(&cfie,0,sizeof(cfie));
137  cfie.cbSize = sizeof(cfie);
138  if(!pGetCurrentConsoleFontEx(hOut,FALSE,&cfie)){
139  handle_api_failure("WgxPrintUnicodeString","GetCurrentConsoleFontEx");
140  } else {
141  if((cfie.FontFamily & WIN_TMPF_TRUETYPE) == WIN_TMPF_TRUETYPE){
142  if(!SetConsoleOutputCP(CP_UTF8)){
143  handle_api_failure("WgxPrintUnicodeString","SetConsoleOutputCP");
144  } else {
145  new_code_page = CP_UTF8;
146  }
147  }
148  }
149  }
150  }
151 
152  /* if the console runs inside of the PowerShell ISE, UTF-8 can be used too */
153  /* however, this approach failed being tested on a Win7 driven machine */
154 
155 convert:
156  /* convert string to the acceptable encoding */
157  length = (wcslen(string) + 1) * 4; /* should be enough */
158  cnv_string = malloc(length);
159  if(cnv_string == NULL){
160  WgxDbgPrint("WgxPrintUnicodeString: not enough memory");
161  goto done;
162  }
163  if(!WideCharToMultiByte(new_code_page,0,string,-1,cnv_string,length,NULL,NULL)){
164  handle_api_failure("WgxPrintUnicodeString","WideCharToMultiByte");
165  goto done;
166  }
167 
168  /* print the converted string */
169  fprintf(f,"%s",cnv_string);
170 
171 done:
172  /* cleanup */
173  SetConsoleOutputCP(old_code_page);
174  free(cnv_string);
175 }
