53 LRESULT CALLBACK NewWndProc(HWND hWnd, UINT iMsg, WPARAM wParam, LPARAM lParam)
54 {
55  int i, found = 0;
56  MSG message;
57 
58  /* search for our window in win array */
59  for(i = 0; i <= idx; i++){
60  if(win[i].hWindow == hWnd){
61  found = 1; break;
62  }
63  }
64 
65  if(found){
66  if(iMsg == WM_KEYDOWN){
67  message.hwnd = hWnd;
68  message.message = iMsg;
69  message.wParam = wParam;
70  message.lParam = lParam;
71  message.pt.x = message.pt.y = 0;
72  message.time = 0;
73  /* if we'll receive a report about improper accelerators work
74  we may use TranslateAcceleratorW to solve the problem;
75  but currently everything seems to be working fine */
76  (void)TranslateAccelerator(win[i].hMainWindow,win[i].hAccelerator,&message);
77  }
78  if(win[i].isWindowUnicode)
79  return CallWindowProcW(win[i].OldWindowProcedure,hWnd,iMsg,wParam,lParam);
80  else
81  return CallWindowProc(win[i].OldWindowProcedure,hWnd,iMsg,wParam,lParam);
82  }
83  /* very extraordinary situation */
84  if(!error_flag){ /* show message box once */
85  error_flag = TRUE;
86  MessageBox(NULL,"OldWindowProcedure is lost!","Error!",MB_OK | MB_ICONHAND);
87  }
88  return 0;
89 }

102 BOOL WgxAddAccelerators(HINSTANCE hInstance,HWND hWindow,UINT AccelId)
103 {
104  HACCEL hAccel;
105  HANDLE hChild;
106  WNDPROC OldWndProc;
107  BOOL isWindowUnicode;
108 
109  /* Load the accelerator table. */
110  hAccel = LoadAccelerators(hInstance,MAKEINTRESOURCE(AccelId));
111  if(!hAccel) return FALSE;
112 
113  if(first_call){
114  memset(win,0,sizeof(win));
115  idx = 0;
116  first_call = FALSE;
117  }
118 
119  /* No need to set accelerator for the main window. */
120  /* Set accelerator for children. */
121  hChild = GetWindow(hWindow,GW_CHILD);
122  while(hChild){
123  if(idx >= (WIN_ARRAY_SIZE - 1))
124  return FALSE; /* too many child windows */
125 
126  if(IsWindowUnicode(hChild)) isWindowUnicode = TRUE;
127  else isWindowUnicode = FALSE;
128 
129  if(isWindowUnicode)
130  OldWndProc = (WNDPROC)SetWindowLongPtrW(hChild,GWLP_WNDPROC,(LONG_PTR)NewWndProc);
131  else
132  OldWndProc = (WNDPROC)SetWindowLongPtr(hChild,GWLP_WNDPROC,(LONG_PTR)NewWndProc);
133 
134  if(OldWndProc){
135  win[idx].hWindow = hChild;
136  win[idx].OldWindowProcedure = OldWndProc;
137  win[idx].hAccelerator = hAccel;
138  win[idx].hMainWindow = hWindow;
139  win[idx].isWindowUnicode = isWindowUnicode;
140  idx ++;
141  }
142  hChild = GetWindow(hChild,GW_HWNDNEXT);
143  }
144 
145  return TRUE;
146 }
