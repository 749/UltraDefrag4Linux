69 BOOL WgxBuildResourceTable(PWGX_I18N_RESOURCE_ENTRY table,char *path)
70 {
71  lua_State *L;
72  int status;
73  char *script;
74  const char *sctemplate = ""
75  "BOM = string.char(0xEF,0xBB,0xBF); "
76  "f = assert(io.open(\"%s\",\"r\")); "
77  "for line in f:lines() do "
78  " i, j, pair = string.find(line,string.format(\"^%%s(.-)$\",BOM)); "
79  " if not pair then pair = line end; "
80  " i, j, key, value = string.find(pair,\"^%%s*(.-)%%s*=%%s*(.-)%%s*$\"); "
81  " if key and value then "
82  " _G[key] = string.gsub(value,\"\\\\n\",\"\\n\") "
83  " end "
84  "end; "
85  "f:close()";
86  int length;
87  const char *msg;
88  int i;
89  const char *value;
90 
91  /* parameters validation */
92  if(!table || !path)
93  return FALSE;
94 
95  L = lua_open();
96  if(L == NULL){
97  WgxDbgPrint("WgxBuildResourceTable: cannot initialize Lua library");
98  return FALSE;
99  }
100 
101  /* stop collector during initialization */
102  lua_gc(L, LUA_GCSTOP, 0);
103  luaL_openlibs(L);
104  lua_gc(L, LUA_GCRESTART, 0);
105 
106  /* prepare Lua script */
107  length = strlen(sctemplate) + strlen(path) + 1;
108  script = malloc(length);
109  if(script == NULL){
110  WgxDbgPrint("WgxBuildResourceTable: not enough memory");
111  lua_close(L);
112  return FALSE;
113  }
114  _snprintf(script,length,sctemplate,path);
115  script[length-1] = 0;
116 
117  /* extract strings from the file */
118  status = luaL_dostring(L, script);
119  if(status){
120  WgxDbgPrint("WgxBuildResourceTable: script execution failed");
121  if(!lua_isnil(L, -1)){
122  msg = lua_tostring(L, -1);
123  if(msg == NULL) msg = "(error object is not a string)";
124  WgxDbgPrint("WgxBuildResourceTable: %s",msg);
125  lua_pop(L, 1);
126  }
127  lua_close(L);
128  free(script);
129  return FALSE;
130  }
131 
132  /* fill the table */
133  if(hSynchEvent){
134  if(WaitForSingleObject(hSynchEvent,INFINITE) != WAIT_OBJECT_0){
135  WgxDbgPrintLastError("WgxBuildResourceTable: synchronization failed");
136  } else {
137  for(i = 0; table[i].Key; i++){
138  lua_getglobal(L, table[i].Key);
139  if(!lua_isnil(L, lua_gettop(L))){
140  value = lua_tostring(L, lua_gettop(L));
141  if(value){
142  length = strlen(value);
143  table[i].LoadedString = malloc((length + 1) * 2);
144  if(table[i].LoadedString == NULL){
145  WgxDbgPrint("WgxBuildResourceTable: not enough memory");
146  } else {
147  if(!MultiByteToWideChar(CP_UTF8,0,value,-1,table[i].LoadedString,length + 1)){
148  WgxDbgPrintLastError("WgxBuildResourceTable: MultiByteToWideChar failed");
149  free(table[i].LoadedString);
150  table[i].LoadedString = NULL;
151  }
152  }
153  }
154  }
155  lua_pop(L, 1);
156  }
157  /* end of synchronization */
158  SetEvent(hSynchEvent);
159  }
160  }
161 
162  lua_close(L);
163  free(script);
164  return TRUE;
165 }

172 void WgxApplyResourceTable(PWGX_I18N_RESOURCE_ENTRY table,HWND hWindow)
173 {
174  int i;
175  HWND hChild;
176  wchar_t *text = NULL;
177 
178  if(!table || !hWindow) return;
179 
180  for(i = 0; table[i].Key; i++){
181  hChild = GetDlgItem(hWindow,table[i].ControlID);
182  /* synchronize access to the LoadedString */
183  if(hSynchEvent){
184  if(WaitForSingleObject(hSynchEvent,INFINITE) != WAIT_OBJECT_0){
185  WgxDbgPrintLastError("WgxApplyResourceTable: synchronization failed");
186  } else {
187  if(table[i].LoadedString){
188  text = _wcsdup(table[i].LoadedString);
189  if(text == NULL)
190  WgxDbgPrint("WgxApplyResourceTable: cannot allocate memory");
191  }
192  /* end of synchronization */
193  SetEvent(hSynchEvent);
194  }
195  }
196  if(text == NULL){
197  text = _wcsdup(table[i].DefaultString);
198  if(text == NULL)
199  WgxDbgPrint("WgxApplyResourceTable: cannot allocate memory");
200  }
201  if(text){
202  (void)SetWindowTextW(hChild,text);
203  free(text);
204  }
205  }
206 }

231 wchar_t *WgxGetResourceString(PWGX_I18N_RESOURCE_ENTRY table,char *key)
232 {
233  int i;
234  wchar_t *text = NULL;
235 
236  if(!table || !key)
237  return NULL;
238 
239  /* synchronize access to the table */
240  if(hSynchEvent){
241  if(WaitForSingleObject(hSynchEvent,INFINITE) != WAIT_OBJECT_0){
242  WgxDbgPrintLastError("WgxDestroyResourceTable: synchronization failed");
243  goto synch_failed;
244  } else {
245  for(i = 0; table[i].Key; i++){
246  if(!strcmp(table[i].Key,key)){
247  if(table[i].LoadedString)
248  text = _wcsdup(table[i].LoadedString);
249  else
250  text = _wcsdup(table[i].DefaultString);
251  if(text == NULL)
252  WgxDbgPrint("WgxGetResourceString: cannot allocate memory");
253  break;
254  }
255  }
256  /* end of synchronization */
257  SetEvent(hSynchEvent);
258  }
259  } else {
260 synch_failed:
261  for(i = 0; table[i].Key; i++){
262  if(!strcmp(table[i].Key,key)){
263  text = _wcsdup(table[i].DefaultString);
264  if(text == NULL)
265  WgxDbgPrint("WgxGetResourceString: cannot allocate memory");
266  break;
267  }
268  }
269  }
270  return text;
271 }
