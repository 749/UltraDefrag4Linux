148 int CreateMainMenu(void)
149 {
150  HBITMAP hBitmap = NULL;
151  HBITMAP hBitmapMasked = NULL;
152  OSVERSIONINFO osvi;
153  int cx, id;
154 
155  WgxDbgPrint("Menu row height ........ %d",GetSystemMetrics(SM_CYMENU));
156  WgxDbgPrint("Menu button size ....... %d x %d",GetSystemMetrics(SM_CXMENUSIZE),GetSystemMetrics(SM_CYMENUSIZE));
157  WgxDbgPrint("Menu check-mark size ... %d x %d",GetSystemMetrics(SM_CXMENUCHECK),GetSystemMetrics(SM_CYMENUCHECK));
158 
159  ZeroMemory(&osvi, sizeof(OSVERSIONINFO));
160  osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
161  GetVersionEx(&osvi);
162  /* menu icons look nicely on Vista and above */
163  if(osvi.dwMajorVersion > 5 && show_menu_icons){
164  cx = GetSystemMetrics(SM_CXMENUCHECK);
165  if(cx < 16){
166  /* 100% DPI */
167  id = IDB_MENU_ICONS_15;
168  } else if(cx < 20){
169  /* 125% DPI */
170  id = IDB_MENU_ICONS_19;
171  } else if(cx < 26){
172  /* 150% DPI */
173  id = IDB_MENU_ICONS_25;
174  } else {
175  /* 200% DPI */
176  id = IDB_MENU_ICONS_31;
177  }
178  hBitmap = LoadBitmap(hInstance, MAKEINTRESOURCE(id));
179  hBitmapMasked = WgxCreateMenuBitmapMasked(hBitmap, (COLORREF)-1);
180  hMainMenu = WgxBuildMenu(main_menu,hBitmapMasked);
181  if(hBitmap != NULL)
182  DeleteObject(hBitmap);
183  if(hBitmapMasked != NULL)
184  DeleteObject(hBitmapMasked);
185  } else {
186  /* on systems below Vista avoid icons use */
187  hMainMenu = WgxBuildMenu(main_menu,NULL);
188  }
189 
190  /* attach menu to the window */
191  if(!SetMenu(hWindow,hMainMenu)){
192  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,
193  "Cannot set main menu!");
194  DestroyMenu(hMainMenu);
195  return (-1);
196  }
197 
198  if(skip_removable == 0){
199  CheckMenuItem(hMainMenu,
200  IDM_IGNORE_REMOVABLE_MEDIA,
201  MF_BYCOMMAND | MF_UNCHECKED);
202  }
203 
204  if(repeat_action){
205  CheckMenuItem(hMainMenu,
206  IDM_REPEAT_ACTION,
207  MF_BYCOMMAND | MF_CHECKED);
208  }
209 
210  if(btd_installed){
211  if(IsBootTimeDefragEnabled()){
212  boot_time_defrag_enabled = 1;
213  CheckMenuItem(hMainMenu,
214  IDM_CFG_BOOT_ENABLE,
215  MF_BYCOMMAND | MF_CHECKED);
216  }
217  } else {
218  EnableMenuItem(hMainMenu,IDM_CFG_BOOT_ENABLE,MF_BYCOMMAND | MF_GRAYED);
219  EnableMenuItem(hMainMenu,IDM_CFG_BOOT_SCRIPT,MF_BYCOMMAND | MF_GRAYED);
220  }
221 
222  if(job_flags & UD_PREVIEW_MATCHING){
223  CheckMenuItem(hMainMenu,
224  IDM_PREVIEW_MATCHING,
225  MF_BYCOMMAND | MF_CHECKED);
226  } else {
227  CheckMenuItem(hMainMenu,
228  IDM_PREVIEW_LARGEST,
229  MF_BYCOMMAND | MF_CHECKED);
230  }
231 
232  if(!DrawMenuBar(hWindow))
233  WgxDbgPrintLastError("Cannot redraw main menu");
234 
235  return 0;
236 }
