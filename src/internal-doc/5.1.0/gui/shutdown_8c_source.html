84  }

86  use_custom_font_in_dialogs ? wgxFont.hFont : NULL,

104  WgxDbgPrint("ResizeShutdownConfirmDialog: cannot get Yes button text");
105  goto done;
106  }
107  text[sizeof(text)/sizeof(wchar_t) - 1] = 0;
108  result = WgxGetTextDimensions(text,

219 BOOL CALLBACK ShutdownConfirmDlgProc(HWND hWnd,UINT msg,WPARAM wParam,LPARAM lParam)
220 {
221  static UINT_PTR timer;
222  static UINT counter;
223  #define TIMER_ID 0x16748382
224  #define MAX_TEXT_LENGTH 127
225  static wchar_t buffer[MAX_TEXT_LENGTH + 1];
226  static wchar_t counter_msg[MAX_TEXT_LENGTH + 1];
227  wchar_t *text = NULL;
228 
229  switch(msg){
230  case WM_INITDIALOG:
231  if(use_custom_font_in_dialogs)
232  WgxSetFont(hWnd,&wgxFont);
233  WgxSetText(hWnd,i18n_table,"PLEASE_CONFIRM");
234  WgxSetText(GetDlgItem(hWnd,IDC_YES_BUTTON),i18n_table,"YES");
235  WgxSetText(GetDlgItem(hWnd,IDC_NO_BUTTON),i18n_table,"NO");
236  switch(when_done_action){
237  case IDM_WHEN_DONE_HIBERNATE:
238  text = WgxGetResourceString(i18n_table,"SECONDS_TILL_HIBERNATION");
239  break;
240  case IDM_WHEN_DONE_LOGOFF:
241  text = WgxGetResourceString(i18n_table,"SECONDS_TILL_LOGOFF");
242  break;
243  case IDM_WHEN_DONE_REBOOT:
244  text = WgxGetResourceString(i18n_table,"SECONDS_TILL_REBOOT");
245  break;
246  case IDM_WHEN_DONE_SHUTDOWN:
247  text = WgxGetResourceString(i18n_table,"SECONDS_TILL_SHUTDOWN");
248  break;
249  }
250  if(text){
251  wcsncpy(counter_msg,text,MAX_TEXT_LENGTH);
252  counter_msg[MAX_TEXT_LENGTH] = 0;
253  free(text);
254  } else {
255  counter_msg[0] = 0;
256  }
257 
258  /* set timer */
259  counter = seconds_for_shutdown_rejection;
260  if(counter == 0)
261  (void)EndDialog(hWnd,1);
262  timer = (UINT_PTR)SetTimer(hWnd,TIMER_ID,1000,NULL);
263  if(timer == 0){
264  /* message box will prevent shutdown which is dangerous */
265  WgxDbgPrintLastError("ShutdownConfirmDlgProc: SetTimer failed");
266  /* force shutdown to avoid situation when pc works a long time without any control */
267  (void)EndDialog(hWnd,1);
268  }
269 
270  /* adjust positions of controls to make the window nice looking */
271  ResizeShutdownConfirmDialog(hWnd,counter_msg);
272 
273  /* shutdown will be rejected by pressing the space key */
274  (void)SetFocus(GetDlgItem(hWnd,IDC_NO_BUTTON));
275 
276  /* bring dialog to foreground */
277  SetForegroundWindow(hWnd);
278  return FALSE;
279  case WM_TIMER:
280  counter --;
281  _snwprintf(buffer,MAX_TEXT_LENGTH + 1,L"%u %ls",counter,counter_msg);
282  buffer[MAX_TEXT_LENGTH] = 0;
283  SetWindowTextW(GetDlgItem(hWnd,IDC_DELAY_MSG),buffer);
284  if(counter == 0){
285  (void)KillTimer(hWnd,TIMER_ID);
286  (void)EndDialog(hWnd,1);
287  }
288  break;
289  case WM_COMMAND:
290  switch(LOWORD(wParam)){
291  case IDC_YES_BUTTON:
292  (void)KillTimer(hWnd,TIMER_ID);
293  (void)EndDialog(hWnd,1);
294  break;
295  case IDC_NO_BUTTON:
296  (void)KillTimer(hWnd,TIMER_ID);
297  (void)EndDialog(hWnd,0);
298  break;
299  }
300  return TRUE;
301  case WM_CLOSE:
302  (void)KillTimer(hWnd,TIMER_ID);
303  (void)EndDialog(hWnd,0);
304  return TRUE;
305  }
306  return FALSE;
307 }

314 int ShutdownOrHibernate(void)
315 {
316  HANDLE hToken;
317  TOKEN_PRIVILEGES tkp;
318  HMODULE hPowrProfDll;
319  SET_SUSPEND_STATE_PROC pSetSuspendState = NULL;
320  BOOL shutdown_cmd_present;
321  BOOL result;
322 
323  switch(when_done_action){
324  case IDM_WHEN_DONE_HIBERNATE:
325  case IDM_WHEN_DONE_LOGOFF:
326  case IDM_WHEN_DONE_REBOOT:
327  case IDM_WHEN_DONE_SHUTDOWN:
328  if(seconds_for_shutdown_rejection){
329  if(DialogBoxW(hInstance,MAKEINTRESOURCEW(IDD_SHUTDOWN),NULL,(DLGPROC)ShutdownConfirmDlgProc) == 0)
330  return 0;
331  /* in case of errors we'll shutdown anyway */
332  /* to avoid situation when pc works a long time without any control */
333  }
334  break;
335  case IDM_WHEN_DONE_EXIT:
336  return 0; /* nothing to do */
337  }
338 
339  /* set SE_SHUTDOWN privilege */
340  if(!OpenProcessToken(GetCurrentProcess(),
341  TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY,&hToken)){
342  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"ShutdownOrHibernate: cannot open process token!");
343  return 4;
344  }
345 
346  LookupPrivilegeValue(NULL,SE_SHUTDOWN_NAME,&tkp.Privileges[0].Luid);
347  tkp.PrivilegeCount = 1; // one privilege to set
348  tkp.Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;
349  AdjustTokenPrivileges(hToken,FALSE,&tkp,0,(PTOKEN_PRIVILEGES)NULL,0);
350  if(GetLastError() != ERROR_SUCCESS){
351  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"ShutdownOrHibernate: cannot set shutdown privilege!");
352  return 5;
353  }
354 
355  /*
356  * Save log file before any action.
357  */
358  udefrag_flush_dbg_log();
359 
360 
361  /*
362  * Shutdown command works better on remote
363  * computers since it shows no confirmation.
364  */
365  shutdown_cmd_present = WgxCreateProcess("%windir%\\system32\\shutdown.exe","/?");
366 
367  /*
368  * There is an opinion that SetSuspendState call
369  * is more reliable than SetSystemPowerState:
370  * http://msdn.microsoft.com/en-us/library/aa373206%28VS.85%29.aspx
371  * On the other hand, it is missing on NT4.
372  */
373  if(when_done_action == IDM_WHEN_DONE_STANDBY || when_done_action == IDM_WHEN_DONE_HIBERNATE){
374  hPowrProfDll = LoadLibrary("powrprof.dll");
375  if(hPowrProfDll == NULL){
376  WgxDbgPrintLastError("ShutdownOrHibernate: cannot load powrprof.dll");
377  } else {
378  pSetSuspendState = (SET_SUSPEND_STATE_PROC)GetProcAddress(hPowrProfDll,"SetSuspendState");
379  if(pSetSuspendState == NULL)
380  WgxDbgPrintLastError("ShutdownOrHibernate: cannot get SetSuspendState address inside powrprof.dll");
381  }
382  if(pSetSuspendState == NULL)
383  WgxDbgPrint("Therefore SetSystemPowerState API will be used instead of SetSuspendState.\n");
384  }
385 
386  switch(when_done_action){
387  case IDM_WHEN_DONE_STANDBY:
388  /* suspend, request permission from apps and drivers */
389  if(pSetSuspendState)
390  result = pSetSuspendState(FALSE,FALSE,FALSE);
391  else
392  result = SetSystemPowerState(TRUE,FALSE);
393  if(!result){
394  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"UltraDefrag: cannot suspend the system!");
395  return 6;
396  }
397  break;
398  case IDM_WHEN_DONE_HIBERNATE:
399  /* hibernate, request permission from apps and drivers */
400  if(pSetSuspendState){
401  result = pSetSuspendState(TRUE,FALSE,FALSE);
402  } else {
403  /* the second parameter must be FALSE, dmitriar's windows xp hangs otherwise */
404  result = SetSystemPowerState(FALSE,FALSE);
405  }
406  if(!result){
407  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"UltraDefrag: cannot hibernate the computer!");
408  return 7;
409  }
410  break;
411  case IDM_WHEN_DONE_LOGOFF:
412  if(shutdown_cmd_present){
413  result = WgxCreateProcess("%windir%\\system32\\cmd.exe","/K shutdown -l");
414  } else {
415  result = ExitWindowsEx(EWX_LOGOFF | EWX_FORCEIFHUNG,
416  SHTDN_REASON_MAJOR_OTHER | SHTDN_REASON_MINOR_OTHER | \
417  SHTDN_REASON_FLAG_PLANNED);
418  }
419  if(!result){
420  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"UltraDefrag: cannot log the user off!");
421  return 8;
422  }
423  break;
424  case IDM_WHEN_DONE_REBOOT:
425  if(shutdown_cmd_present){
426  result = WgxCreateProcess("%windir%\\system32\\cmd.exe","/K shutdown -r -t 0");
427  } else {
428  result = ExitWindowsEx(EWX_REBOOT | EWX_FORCEIFHUNG,
429  SHTDN_REASON_MAJOR_OTHER | SHTDN_REASON_MINOR_OTHER | \
430  SHTDN_REASON_FLAG_PLANNED);
431  }
432  if(!result){
433  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"UltraDefrag: cannot reboot the computer!");
434  return 9;
435  }
436  break;
437  case IDM_WHEN_DONE_SHUTDOWN:
438  if(shutdown_cmd_present){
439  result = WgxCreateProcess("%windir%\\system32\\cmd.exe","/K shutdown -s -t 0");
440  } else {
441  result = ExitWindowsEx(EWX_POWEROFF | EWX_FORCEIFHUNG,
442  SHTDN_REASON_MAJOR_OTHER | SHTDN_REASON_MINOR_OTHER | \
443  SHTDN_REASON_FLAG_PLANNED);
444  }
445  if(!result){
446  WgxDisplayLastError(NULL,MB_OK | MB_ICONHAND,"UltraDefrag: cannot shut down the computer!");
447  return 10;
448  }
449  break;
450  }
451 
452  return 0;
453 }
