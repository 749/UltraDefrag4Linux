107 static void update_progress(udefrag_progress_info *pi, void *p)
108 {
109  volume_processing_job *job;
110  char WindowCaption[256];
111  char current_operation;
112 
113  job = current_job;
114 
115  if(job == NULL)
116  return;
117 
118  memcpy(&job->pi,pi,sizeof(udefrag_progress_info));
119 
120  VolListUpdateStatusField(job);
121  VolListUpdateFragmentationField(job);
122  UpdateStatusBar(pi);
123 
124  switch(pi->current_operation){
125  case VOLUME_ANALYSIS:
126  current_operation = 'A';
127  break;
128  case VOLUME_DEFRAGMENTATION:
129  current_operation = 'D';
130  break;
131  default:
132  current_operation = 'O';
133  break;
134  }
135 
136  if(dry_run){
137  (void)sprintf(WindowCaption, "%c: %c %6.2lf %% (dry run)",
138  udefrag_toupper(job->volume_letter), current_operation, pi->percentage);
139  } else {
140  (void)sprintf(WindowCaption, "%c: %c %6.2lf %%",
141  udefrag_toupper(job->volume_letter), current_operation, pi->percentage);
142  }
143  (void)SetWindowText(hWindow, WindowCaption);
144 
145  if(WaitForSingleObject(hMapEvent,INFINITE) != WAIT_OBJECT_0){
146  WgxDbgPrintLastError("update_progress: wait on hMapEvent failed");
147  return;
148  }
149  if(pi->cluster_map){
150  if(job->map.buffer == NULL || pi->cluster_map_size != job->map.size){
151  free(job->map.buffer);
152  job->map.buffer = malloc(pi->cluster_map_size);
153  }
154  if(job->map.buffer == NULL){
155  WgxDbgPrint("update_progress: cannot allocate %u bytes of memory\n",
156  pi->cluster_map_size);
157  job->map.size = 0;
158  } else {
159  job->map.size = pi->cluster_map_size;
160  memcpy(job->map.buffer,pi->cluster_map,pi->cluster_map_size);
161  }
162  }
163  SetEvent(hMapEvent);
164 
165  if(pi->cluster_map && job->map.buffer)
166  RedrawMap(job,1);
167 
168  if(pi->completion_status != 0/* && !stop_pressed*/){
169  if(dry_run == 0){
170  if(portable_mode) SetWindowText(hWindow,VERSIONINTITLE_PORTABLE);
171  else SetWindowText(hWindow,VERSIONINTITLE);
172  } else {
173  if(portable_mode) SetWindowText(hWindow,VERSIONINTITLE_PORTABLE " (dry run)");
174  else SetWindowText(hWindow,VERSIONINTITLE " (dry run)");
175  }
176  }
177 }

241 static void ProcessSingleVolume(volume_processing_job *job)
242 {
243  volume_info v;
244  int error_code;
245  int index;
246 
247  if(job == NULL)
248  return;
249 
250  /* refresh capacity information of the volume (bug #2036873) */
251  VolListRefreshItem(job);
252  /*ShowProgress();
253  SetProgress(L"A 0.00 %",0);*/
254 
255  /* validate the volume before any processing */
256  error_code = udefrag_validate_volume(job->volume_letter,FALSE);
257  if(error_code < 0){
258  /* handle error */
259  DisplayInvalidVolumeError(error_code);
260  } else {
261  /* process the volume */
262  current_job = job;
263  if(repeat_action) job_flags |= UD_JOB_REPEAT;
264  else job_flags &= ~UD_JOB_REPEAT;
265  error_code = udefrag_start_job(job->volume_letter, job->job_type,
266  job_flags,map_blocks_per_line * map_lines,update_progress,
267  terminator, NULL);
268  if(error_code < 0 && !exit_pressed){
269  DisplayDefragError(error_code);
270  }
271  /* update dirty volume mark */
272  index = get_job_index(job);
273  if(index != -1){
274  if(udefrag_get_volume_information(job->volume_letter,&v) >= 0)
275  SetVolumeDirtyStatus(index,&v);
276  }
277  }
