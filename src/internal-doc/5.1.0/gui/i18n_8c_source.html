223 void ApplyLanguagePack(void)
224 {
225  char lang_name[MAX_PATH];
226  char path[MAX_PATH];
227  udefrag_progress_info pi;
228  MENUITEMINFOW mi;
229  wchar_t *text;
230  wchar_t *s = L"";
231  wchar_t buffer[256];
232  int i;
233  LVCOLUMNW lvc;
234 
235  /* read lang.ini file */
236  GetPrivateProfileString("Language","Selected","",lang_name,MAX_PATH,".\\lang.ini");
237  if(lang_name[0] == 0){
238  WgxDbgPrint("Selected language name not found in lang.ini file\n");
239  /* assign default strings to the toolbar tooltips */
240  UpdateToolbarTooltips();
241  return;
242  }
243  _snprintf(path,MAX_PATH,".\\\\i18n\\\\%s.lng",lang_name);
244  path[MAX_PATH - 1] = 0;
245 
246  /* destroy resource table */
247  WgxDestroyResourceTable(i18n_table);
248 
249  /* build new resource table */
250  WgxBuildResourceTable(i18n_table,path);
251 
252  /* apply new strings to the list of volumes */
253  lvc.mask = LVCF_TEXT;
254  lvc.pszText = text = WgxGetResourceString(i18n_table,"VOLUME");
255  if(text){
256  SendMessage(hList,LVM_SETCOLUMNW,0,(LPARAM)&lvc);
257  free(text);
258  }
259  lvc.pszText = text = WgxGetResourceString(i18n_table,"STATUS");
260  if(text){
261  SendMessage(hList,LVM_SETCOLUMNW,1,(LPARAM)&lvc);
262  free(text);
263  }
264  lvc.pszText = text = WgxGetResourceString(i18n_table,"FRAGMENTATION");
265  if(text){
266  SendMessage(hList,LVM_SETCOLUMNW,2,(LPARAM)&lvc);
267  free(text);
268  }
269  lvc.pszText = text = WgxGetResourceString(i18n_table,"TOTAL");
270  if(text){
271  SendMessage(hList,LVM_SETCOLUMNW,3,(LPARAM)&lvc);
272  free(text);
273  }
274  lvc.pszText = text = WgxGetResourceString(i18n_table,"FREE");
275  if(text){
276  SendMessage(hList,LVM_SETCOLUMNW,4,(LPARAM)&lvc);
277  free(text);
278  }
279  lvc.pszText = text = WgxGetResourceString(i18n_table,"PERCENT");
280  if(text){
281  SendMessage(hList,LVM_SETCOLUMNW,5,(LPARAM)&lvc);
282  free(text);
283  }
284 
285  /* apply new strings to the main menu */
286  for(i = 0; menu_items[i].id; i++){
287  s = WgxGetResourceString(i18n_table,menu_items[i].key);
288  if(s){
289  if(menu_items[i].hotkeys)
290  _snwprintf(buffer,256,L"%ws\t%hs",s,menu_items[i].hotkeys);
291  else
292  _snwprintf(buffer,256,L"%ws",s);
293  buffer[255] = 0;
294  memset(&mi,0,MENUITEMINFOW_SIZE);
295  mi.cbSize = MENUITEMINFOW_SIZE;
296  mi.fMask = MIIM_TYPE;
297  mi.fType = MFT_STRING;
298  mi.dwTypeData = buffer;
299  SetMenuItemInfoW(hMainMenu,menu_items[i].id,FALSE,&mi);
300  free(s);
301  }
302  }
303 
304  /* apply new strings to the toolbar */
305  UpdateToolbarTooltips();
306 
307  /* redraw main menu */
308  if(!DrawMenuBar(hWindow))
309  WgxDbgPrintLastError("Cannot redraw main menu");
310 
311  /* refresh volume status fields */
312  update_status_of_all_jobs();
313 
314  /* refresh status bar */
315  if(current_job){
316  UpdateStatusBar(&current_job->pi);
317  } else {
318  memset(&pi,0,sizeof(udefrag_progress_info));
319  UpdateStatusBar(&pi);
320  }
321 
322  /* update taskbar icon overlay */
323  if(show_taskbar_icon_overlay){
324  if(WaitForSingleObject(hTaskbarIconEvent,INFINITE) != WAIT_OBJECT_0){
325  WgxDbgPrintLastError("ApplyLanguagePack: wait on hTaskbarIconEvent failed");
326  } else {
327  RemoveTaskbarIconOverlay();
328  if(job_is_running)
329  SetTaskbarIconOverlay(IDI_BUSY,"JOB_IS_RUNNING");
330  SetEvent(hTaskbarIconEvent);
331  }
332  }
333 
334  /* define whether to use custom font for dialog boxes or not */
335  use_custom_font_in_dialogs = 0;
336  for(i = 0; special_font_languages[i]; i++){
337  if(strcmp(lang_name,special_font_languages[i]) == 0){
338  use_custom_font_in_dialogs = 1;
339  break;
340  }
341  }
342 }

364 void BuildLanguageMenu(void)
365 {
366  MENUITEMINFO mi;
367  wchar_t *text;
368  HMENU hLangMenu;
369  intptr_t h;
370  struct _wfinddata_t lng_file;
371  wchar_t filename[MAX_PATH];
372  int i, length;
373  wchar_t selected_lang_name[MAX_PATH];
374  UINT flags;
375 
376  struct prb_table *pt;
377  struct prb_traverser t;
378  wchar_t *f;
379 
380  /* synchronize with other threads */
381  if(WaitForSingleObject(hLangMenuEvent,INFINITE) != WAIT_OBJECT_0){
382  WgxDbgPrintLastError("BuildLanguageMenu: wait on hLangMenuEvent failed");
383  return;
384  }
385 
386  /* get selected language name */
387  GetPrivateProfileStringW(L"Language",L"Selected",L"",selected_lang_name,MAX_PATH,L".\\lang.ini");
388  if(selected_lang_name[0] == 0)
389  wcscpy(selected_lang_name,L"English (US)");
390 
391  /* get submenu handle */
392  memset(&mi,0,MENUITEMINFO_SIZE);
393  mi.cbSize = MENUITEMINFO_SIZE;
394  mi.fMask = MIIM_SUBMENU;
395  if(!GetMenuItemInfo(hMainMenu,IDM_LANGUAGE,FALSE,&mi)){
396  WgxDbgPrintLastError("BuildLanguageMenu: cannot get submenu handle");
397  SetEvent(hLangMenuEvent);
398  return;
399  }
400  hLangMenu = mi.hSubMenu;
401 
402  /* detach submenu */
403  memset(&mi,0,MENUITEMINFO_SIZE);
404  mi.cbSize = MENUITEMINFO_SIZE;
405  mi.fMask = MIIM_SUBMENU;
406  mi.hSubMenu = NULL;
407  if(!SetMenuItemInfo(hMainMenu,IDM_LANGUAGE,FALSE,&mi)){
408  WgxDbgPrintLastError("BuildLanguageMenu: cannot detach submenu");
409  SetEvent(hLangMenuEvent);
410  return;
411  }
412 
413  /* destroy submenu */
414  DestroyMenu(hLangMenu);
415 
416  /* build new menu from the list of installed files */
417  hLangMenu = CreatePopupMenu();
418  if(hLangMenu == NULL){
419  WgxDbgPrintLastError("BuildLanguageMenu: cannot create submenu");
420  SetEvent(hLangMenuEvent);
421  return;
422  }
423 
424  /* add translation menu items and a separator */
425  text = WgxGetResourceString(i18n_table,"TRANSLATIONS_CHANGE_LOG");
426  if(text){
427  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_CHANGE_LOG,text))
428  WgxDbgPrintLastError("BuildLanguageMenu: cannot append change log");
429  free(text);
430  } else {
431  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_CHANGE_LOG,L"&View change log"))
432  WgxDbgPrintLastError("BuildLanguageMenu: cannot append change log");
433  }
434  text = WgxGetResourceString(i18n_table,"TRANSLATIONS_REPORT");
435  if(text){
436  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_REPORT,text))
437  WgxDbgPrintLastError("BuildLanguageMenu: cannot append report");
438  free(text);
439  } else {
440  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_REPORT,L"View translation &report"))
441  WgxDbgPrintLastError("BuildLanguageMenu: cannot append report");
442  }
443  text = WgxGetResourceString(i18n_table,"TRANSLATIONS_FOLDER");
444  if(text){
445  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_FOLDER,text))
446  WgxDbgPrintLastError("BuildLanguageMenu: cannot append folder");
447  free(text);
448  } else {
449  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_FOLDER,L"&Translations folder"))
450  WgxDbgPrintLastError("BuildLanguageMenu: cannot append folder");
451  }
452  text = WgxGetResourceString(i18n_table,"TRANSLATIONS_SUBMIT");
453  if(text){
454  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_SUBMIT,text))
455  WgxDbgPrintLastError("BuildLanguageMenu: cannot append submit");
456  free(text);
457  } else {
458  if(!AppendMenuW(hLangMenu,MF_STRING | MF_ENABLED,IDM_TRANSLATIONS_SUBMIT,L"&Submit current translation"))
459  WgxDbgPrintLastError("BuildLanguageMenu: cannot append submit");
460  }
461  AppendMenu(hLangMenu,MF_SEPARATOR,0,NULL);
462 
463  h = _wfindfirst(L".\\i18n\\*.lng",&lng_file);
464  if(h == -1){
465  WgxDbgPrint("BuildLanguageMenu: no language packs found\n");
466 no_files_found:
467  /* add default US English */
468  if(!AppendMenu(hLangMenu,MF_STRING | MF_ENABLED | MF_CHECKED,IDM_LANGUAGE + 0x1,"English (US)")){
469  WgxDbgPrintLastError("BuildLanguageMenu: cannot append menu item");
470  DestroyMenu(hLangMenu);
471  SetEvent(hLangMenuEvent);
472  return;
473  }
474  } else {
475  /* use binary tree to sort items */
476  pt = prb_create(names_compare,NULL,NULL);
477  if(pt == NULL){
478  /* this case is extraordinary */
479  WgxDbgPrint("BuildLanguageMenu: prb_create failed!");
480  _findclose(h);
481  goto no_files_found;
482  }
483  wcsncpy(filename,lng_file.name,MAX_PATH - 1);
484  filename[MAX_PATH - 1] = 0;
485  length = wcslen(filename);
486  if(length > (int)wcslen(L".lng"))
487  filename[length - 4] = 0;
488  f = _wcsdup(filename);
489  if(f == NULL){
490  WgxDbgPrint("BuildLanguageMenu: not enough memory!");
491  } else {
492  if(prb_probe(pt,(void *)f) == NULL){
493  WgxDbgPrint("BuildLanguageMenu: prb_probe failed for %ws!",f);
494  free(f);
495  }
496  }
497  while(_wfindnext(h,&lng_file) == 0){
498  wcsncpy(filename,lng_file.name,MAX_PATH - 1);
499  filename[MAX_PATH - 1] = 0;
500  length = wcslen(filename);
501  if(length > (int)wcslen(L".lng"))
502  filename[length - 4] = 0;
503  f = _wcsdup(filename);
504  if(f == NULL){
505  WgxDbgPrint("BuildLanguageMenu: not enough memory!");
506  } else {
507  if(prb_probe(pt,(void *)f) == NULL){
508  WgxDbgPrint("BuildLanguageMenu: prb_probe failed for %ws!",f);
509  free(f);
510  }
511  }
512  }
513  _findclose(h);
514 
515  /* build the menu */
516  i = 0x1;
517  prb_t_init(&t,pt);
518  f = (wchar_t *)prb_t_first(&t,pt);
519  while(f != NULL){
520  flags = MF_STRING | MF_ENABLED;
521  if(wcscmp(selected_lang_name,f) == 0)
522  flags |= MF_CHECKED;
523  if(!AppendMenuW(hLangMenu,flags,IDM_LANGUAGE + i,f)){
524  WgxDbgPrintLastError("BuildLanguageMenu: cannot append menu item");
525  } else {
526  i++;
527  }
528  f = (wchar_t *)prb_t_next(&t);
529  }
530 
531  /* destroy binary tree */
532  prb_destroy(pt,free_item);
533  }
534 
535  /* attach submenu to the Language menu */
536  memset(&mi,0,MENUITEMINFO_SIZE);
537  mi.cbSize = MENUITEMINFO_SIZE;
538  mi.fMask = MIIM_SUBMENU;
539  mi.hSubMenu = hLangMenu;
540  if(!SetMenuItemInfo(hMainMenu,IDM_LANGUAGE,FALSE,&mi)){
541  WgxDbgPrintLastError("BuildLanguageMenu: cannot attach submenu");
542  DestroyMenu(hLangMenu);
543  SetEvent(hLangMenuEvent);
544  return;
545  }
546 
547  /* end of synchronization */
548  SetEvent(hLangMenuEvent);
549 }
