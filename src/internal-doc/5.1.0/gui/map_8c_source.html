131 void ResizeMap(int x, int y, int width, int height)
132 {
133  RECT rc;
134  int border_width, border_height;
135  long dx, dy, threshold;
136 
137  if(WaitForSingleObject(hMapEvent,INFINITE) != WAIT_OBJECT_0){
138  WgxDbgPrintLastError("ResizeMap: wait on hMapEvent failed");
139  return;
140  }
141 
142  /* get coordinates of the map field, without borders */
143  border_width = GetSystemMetrics(SM_CXEDGE);
144  border_height = GetSystemMetrics(SM_CYEDGE);
145  rc.left = x + border_width;
146  rc.right = (x + width) - border_width;
147  rc.top = y + border_height;
148  rc.bottom = (y + height) - border_height;
149 
150  /* calculate number of blocks and a real size of the map control */
151  map_blocks_per_line = (rc.right - rc.left - grid_line_width) / (map_block_size + grid_line_width);
152  map_width = (map_block_size + grid_line_width) * map_blocks_per_line + grid_line_width;
153  map_lines = (rc.bottom - rc.top - grid_line_width) / (map_block_size + grid_line_width);
154  map_height = (map_block_size + grid_line_width) * map_lines + grid_line_width;
155 
156  /* center the map control */
157  dx = (rc.right - rc.left - map_width) / 2;
158  dy = (rc.bottom - rc.top - map_height) / 2;
159  // comparison with zero causes more flicker of the map
160  threshold = grid_line_width * 2;
161  if(dx > threshold) rc.left += dx;
162  if(dy > threshold) rc.top += dy;
163 
164  last_block_size = map_block_size;
165  last_grid_width = grid_line_width;
166  last_x = x;
167  last_y = y;
168  last_width = width;
169  last_height = height;
170 
171  /* reposition the map control */
172  (void)SetWindowPos(hMap, NULL,
173  rc.left - border_width,
174  rc.top - border_height,
175  map_width + 2 * border_width,
176  map_height + 2 * border_height,
177  SWP_NOZORDER);
178  SetEvent(hMapEvent);
179 }

184 static void DrawGrid(HDC hdc)
185 {
186  HPEN hPen, hOldPen;
187  RECT rc;
188  int i, j;
189 
190  /* draw white field */
191  rc.top = rc.left = 0;
192  rc.bottom = map_height;
193  rc.right = map_width;
194  (void)FillRect(hdc,&rc,hBrushes[FREE_SPACE]);
195 
196  if(grid_line_width == 0)
197  return;
198 
199  grid_color = RGB(grid_color_r,grid_color_g,grid_color_b);
200  hPen = CreatePen(PS_SOLID,1,grid_color);
201  hOldPen = SelectObject(hdc,hPen);
202  /* draw vertical lines */
203  for(i = 0; i < map_blocks_per_line + 1; i++){
204  for(j = 0; j < grid_line_width; j++){
205  (void)MoveToEx(hdc,(map_block_size + grid_line_width) * i + j,0,NULL);
206  (void)LineTo(hdc,(map_block_size + grid_line_width) * i + j,map_height);
207  }
208  }
209  /* draw horizontal lines */
210  for(i = 0; i < map_lines + 1; i++){
211  for(j = 0; j < grid_line_width; j++){
212  (void)MoveToEx(hdc,0,(map_block_size + grid_line_width) * i + j,NULL);
213  (void)LineTo(hdc,map_width,(map_block_size + grid_line_width) * i + j);
214  }
215  }
216  (void)SelectObject(hdc,hOldPen);
217  (void)DeleteObject(hPen);
218 }

273 void RedrawMap(volume_processing_job *job, int map_refill_required)
274 {
275  HDC hdc;
276  HDC hDC, hMainDC;
277  HBITMAP hBitmap;
278  int i, j, k, ratio, color;
279  int array[NUM_OF_SPACE_STATES];
280  int maximum;
281  int mft_detected;
282  int used_cells = 0;
283 
284  if(job == NULL)
285  return;
286 
287  //if(!allow_map_redraw)
288  //return;
289 
290  if(map_blocks_per_line == 0 || map_lines == 0)
291  return;
292 
293  if(WaitForSingleObject(hMapEvent,INFINITE) != WAIT_OBJECT_0){
294  WgxDbgPrintLastError("RedrawMap: wait on hMapEvent failed");
295  return;
296  }
297 
298  /* if background color changed, redefine it */
299  if(colors[FREE_SPACE] != RGB(free_color_r,free_color_g,free_color_b)){
300  (void)DeleteObject(hBrushes[FREE_SPACE]);
301  colors[FREE_SPACE] = RGB(free_color_r,free_color_g,free_color_b);
302  hBrushes[FREE_SPACE] = CreateSolidBrush(colors[FREE_SPACE]);
303  map_refill_required = 1;
304  }
305 
306  /* if volume bitmap is empty or have improper size, recreate it */
307  if(job->map.hdc == NULL || job->map.hbitmap == NULL ||
308  job->map.width != map_width || job->map.height != map_height){
309  if(job->map.hdc){
310  (void)DeleteDC(job->map.hdc);
311  job->map.hdc = NULL;
312  }
313  if(job->map.hbitmap){
314  (void)DeleteObject(job->map.hbitmap);
315  job->map.hbitmap = NULL;
316  }
317  hMainDC = GetDC(hWindow);
318  hDC = CreateCompatibleDC(hMainDC);
319  hBitmap = CreateCompatibleBitmap(hMainDC,map_width,map_height);
320 
321  if(hBitmap == NULL){
322  WgxDbgPrintLastError("RedrawMap: CreateCompatibleBitmap failed");
323  (void)ReleaseDC(hWindow,hMainDC);
324  (void)DeleteDC(hDC);
325  SetEvent(hMapEvent);
326  return;
327  }
328 
329  (void)ReleaseDC(hWindow,hMainDC);
330  (void)SelectObject(hDC,hBitmap);
331  (void)SetBkMode(hDC,TRANSPARENT);
332  DrawGrid(hDC);
333  job->map.hdc = hDC;
334  job->map.hbitmap = hBitmap;
335  job->map.width = map_width;
336  job->map.height = map_height;
337  map_refill_required = 1;
338  } else {
339  /* if block size changed, but map dimensions aren't, redraw grid anyway */
340  if(job->map.scaled_size != map_blocks_per_line * map_lines){
341  DrawGrid(job->map.hdc);
342  map_refill_required = 1;
343  }
344  /* if grid color changed, redraw it */
345  if(grid_color != RGB(grid_color_r,grid_color_g,grid_color_b)){
346  DrawGrid(job->map.hdc);
347  }
348  }
349 
350  /* if cluster map does not exist, draw empty bitmap */
351  if(job->map.buffer == NULL)
352  goto redraw;
353 
354  /* scale map */
355  if(job->map.scaled_buffer == NULL \
356  || job->map.scaled_size != map_blocks_per_line * map_lines){
357  free(job->map.scaled_buffer);
358  job->map.scaled_buffer = malloc(map_blocks_per_line * map_lines);
359  if(job->map.scaled_buffer == NULL){
360  WgxDbgPrint("RedrawMap: cannot allocate %u bytes of memory\n",
361  map_blocks_per_line * map_lines);
362  job->map.scaled_size = 0;
363  SetEvent(hMapEvent);
364  return;
365  } else {
366  job->map.scaled_size = map_blocks_per_line * map_lines;
367  }
368  map_refill_required = 1;
369  }
370 
371  if(job->map.scaled_size == job->map.size){
372  /* no need for scaling, draw directly */
373  if(map_refill_required)
374  FillMap(job);
375  goto redraw;
376  }
377 
378  /* draw scaled map */
379  ratio = job->map.scaled_size / job->map.size;
380  if(ratio != 0){
381  /* scale up */
382  for(i = 0, k = 0; i < job->map.size; i++){
383  for(j = 0; j < ratio; j++){
384  job->map.scaled_buffer[k] = job->map.buffer[i];
385  k++;
386  }
387  }
388  used_cells = k;
389  } else {
390  /* scale down */
391  ratio = job->map.size / job->map.scaled_size;
392  if(ratio * job->map.scaled_size != job->map.size)
393  ratio ++; /* round up */
394  used_cells = job->map.size / ratio;
395  for(i = 0; i < used_cells - 1; i++){
396  /* get dominating color */
397  memset(array,0,sizeof(array));
398  mft_detected = 0;
399  for(j = 0; j < ratio; j++){
400  color = (int)job->map.buffer[i * ratio + j];
401  if(color >= 0 && color < NUM_OF_SPACE_STATES)
402  array[color] ++;
403  if(color == MFT_SPACE) mft_detected = 1;
404  }
405  maximum = array[0], color = 0;
406  for(j = 1; j < NUM_OF_SPACE_STATES; j++){
407  if(array[j] >= maximum){
408  maximum = array[j], color = j;
409  }
410  }
411  /* redraw cell */
412  if(mft_detected){
413  job->map.scaled_buffer[i] = MFT_SPACE;
414  } else {
415  job->map.scaled_buffer[i] = (char)color;
416  }
417  }
418  /* redraw last used cell */
419  memset(array,0,sizeof(array));
420  mft_detected = 0;
421  for(j = (used_cells - 1) * ratio; j < job->map.size; j++){
422  color = (int)job->map.buffer[j];
423  if(color >= 0 && color < NUM_OF_SPACE_STATES)
424  array[color] ++;
425  if(color == MFT_SPACE) mft_detected = 1;
426  }
427  maximum = array[0], color = 0;
428  for(j = 1; j < NUM_OF_SPACE_STATES; j++){
429  if(array[j] >= maximum){
430  maximum = array[j], color = j;
431  }
432  }
433  if(mft_detected){
434  job->map.scaled_buffer[used_cells - 1] = MFT_SPACE;
435  } else {
436  job->map.scaled_buffer[used_cells - 1] = (char)color;
437  }
438  }
439  /* remark unused cells */
440  for(i = used_cells; i < job->map.scaled_size; i++)
441  job->map.scaled_buffer[i] = UNUSED_MAP_SPACE;
442  if(map_refill_required)
443  FillScaledMap(job);
444 
445 redraw:
446  hdc = GetDC(hMap);
447  (void)BitBlt(hdc,0,0,map_width,map_height,job->map.hdc,0,0,SRCCOPY);
448  (void)ReleaseDC(hMap,hdc);
449  SetEvent(hMapEvent);
450 }
