74 DWORD WINAPI CrashInfoCheckingProc(LPVOID lpParameter)
75 {
76  char *msg = "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\r\n"
77  "\r\n"
78  " If this file contains the UltraDefrag crash information\r\n"
79  " you can help to fix the bug which caused the crash\r\n"
80  " by submitting this information to the bug tracker:\r\n"
81  " http://sourceforge.net/tracker/?group_id=199532&atid=969870\r\n"
82  "\r\n"
83  "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\r\n"
84  "\r\n";
85  DWORD bytes_written;
86  char time_stamp[32];
87 
88  /* open crash-info.log file for exclusive access */
89  hLogFile = CreateFile("crash-info.log",GENERIC_WRITE,
90  FILE_SHARE_READ,NULL,CREATE_ALWAYS,FILE_ATTRIBUTE_NORMAL,NULL);
91  if(hLogFile == NULL){
92  WgxDbgPrintLastError("CrashInfoCheckingProc: cannot create crash-info.log file");
93  goto done;
94  }
95 
96  /* write header to the file */
97  if(!WriteFile(hLogFile,msg,strlen(msg),&bytes_written,NULL)){
98  WgxDbgPrintLastError("CrashInfoCheckingProc: cannot write to crash-info.log file");
99  }
100 
101  /* get time stamp of the last processed event */
102  GetPrivateProfileString("LastProcessedEvent","TimeStamp","0",time_stamp,32,".\\crash-info.ini");
103  last_time_stamp = atoi(time_stamp);
104  if(stop_crash_info_check) goto done;
105 
106  /* collect information on all the events not processed yet */
107  CollectCrashInfo();
108  if(stop_crash_info_check) goto done;
109 
110  /* show the collected information */
111  ShowCrashInfo();
112 
113  /* save the time stamp of the last processed event */
114  _snprintf(time_stamp,32,"%u",last_time_stamp);
115  time_stamp[31] = 0;
116  WritePrivateProfileString("LastProcessedEvent","TimeStamp",time_stamp,".\\crash-info.ini");
117 
118 done:
119  if(hLogFile) CloseHandle(hLogFile);
120  crash_info_check_stopped = 1;
121  return 0;
122 }

127 static void CollectCrashInfo(void)
128 {
129  char *buffer = NULL;
130  HANDLE hLog = NULL;
131  DWORD bytes_to_read;
132  DWORD bytes_read;
133  DWORD bytes_needed;
134  BOOL result;
135  DWORD error;
136  PEVENTLOGRECORD rec;
137  char *data;
138  DWORD bytes_written;
139  DWORD new_time_stamp;
140 
141  new_time_stamp = last_time_stamp;
142 
143  bytes_to_read = EVENT_BUFFER_SIZE;
144  buffer = malloc(bytes_to_read);
145  if(buffer == NULL){
146  WgxDbgPrint("CollectCrashInfo: cannot allocate memory");
147  goto done;
148  }
149 
150  hLog = OpenEventLog(NULL,"Application");
151  if(hLog == NULL){
152  WgxDbgPrintLastError("CollectCrashInfo: cannot open the Application event log");
153  goto done;
154  }
155 
156  while(!stop_crash_info_check){
157  result = ReadEventLog(hLog,
158  EVENTLOG_SEQUENTIAL_READ | EVENTLOG_BACKWARDS_READ,
159  0,buffer,bytes_to_read,&bytes_read,&bytes_needed);
160  if(!result){
161  /* handle errors */
162  error = GetLastError();
163  switch(error){
164  case ERROR_INSUFFICIENT_BUFFER:
165  WgxDbgPrint("CollectCrashInfo: larger buffer of %u bytes is needed",bytes_needed);
166  bytes_to_read = bytes_needed;
167  free(buffer);
168  buffer = malloc(bytes_to_read);
169  if(buffer == NULL){
170  WgxDbgPrint("CollectCrashInfo: cannot allocate memory");
171  goto done;
172  }
173  break;
174  case ERROR_HANDLE_EOF:
175  goto done;
176  default:
177  WgxDbgPrintLastError("CollectCrashInfo: ReadEventLog failed");
178  goto done;
179  }
180  } else {
181  /* handle events */
182  rec = (PEVENTLOGRECORD)buffer;
183  while(!stop_crash_info_check){
184  if((char *)rec >= buffer + bytes_read) break;
185  /* skip records older than already processed before */
186  if(rec->TimeGenerated <= last_time_stamp) break;
187  /* handle Application Error events only */
188  if(rec->EventType == EVENTLOG_ERROR_TYPE && (rec->EventID & 0xFFFF) == 1000){
189  if(rec->DataLength > 0){
190  data = malloc(rec->DataLength + 1);
191  if(data == NULL){
192  WgxDbgPrint("CollectCrashInfo: cannot allocate memory for event data");
193  } else {
194  memcpy(data,(char *)rec + rec->DataOffset,rec->DataLength);
195  data[rec->DataLength] = 0;
196  /* handle UltraDefrag GUI and command line tool crashes only */
197  if(strstr(data,"ultradefrag.exe") || strstr(data,"udefrag.exe")){
198  WgxDbgPrint("Crashed in the past: %s",data);
199  if(!WriteFile(hLogFile,data,rec->DataLength,&bytes_written,NULL)){
200  WgxDbgPrintLastError("CollectCrashInfo: cannot write to crash-info.log file");
201  } else {
202  crash_info_collected = 1;
203  if(rec->TimeGenerated > new_time_stamp)
204  new_time_stamp = rec->TimeGenerated;
205  }
206  }
207  free(data);
208  }
209  }
210  }
211  rec = (PEVENTLOGRECORD)((char *)rec + rec->Length);
212  }
213  }
214  }
215 
216 done:
217  last_time_stamp = new_time_stamp;
218  if(hLog) CloseEventLog(hLog);
219  free(buffer);
220 }
