169 void GetPrefs(void)
170 {
171  /*
172  * The program should be configurable
173  * through guiopts.lua file only.
174  */
175  CleanupEnvironment();
176 
177  ValidateGUIOptions();
178  WgxGetOptions(".\\options\\guiopts.lua",read_only_options);
179  WgxGetOptions(".\\options\\guiopts-internals.lua",internal_options);
180 
181  /* get restored main window coordinates */
182  r_rc.left = rx;
183  r_rc.top = ry;
184  r_rc.right = rx + rwidth;
185  r_rc.bottom = ry + rheight;
186 
187  init_maximized_window = maximized_window;
188  if(map_block_size < 0)
189  map_block_size = DEFAULT_MAP_BLOCK_SIZE;
190  if(grid_line_width < 0)
191  grid_line_width = DEFAULT_GRID_LINE_WIDTH;
192 
193  (void)udefrag_set_log_file_path();
194 }

234 DWORD WINAPI PrefsChangesTrackingProc(LPVOID lpParameter)
235 {
236  HANDLE h;
237  DWORD status;
238  RECT rc;
239  int s_maximized, s_init_maximized;
240  int s_skip_removable;
241  int s_repeat_action;
242  int s_job_flags;
243  int cw[sizeof(user_defined_column_widths) / sizeof(int)];
244  int s_list_height;
245  int s_show_taskbar_icon_overlay;
246  ULONGLONG counter = 0;
247 
248  h = FindFirstChangeNotification(".\\options",
249  FALSE,FILE_NOTIFY_CHANGE_LAST_WRITE);
250  if(h == INVALID_HANDLE_VALUE){
251  WgxDbgPrintLastError("PrefsChangesTrackingProc: FindFirstChangeNotification failed");
252  changes_tracking_stopped = 1;
253  return 0;
254  }
255 
256  while(!stop_track_changes){
257  status = WaitForSingleObject(h,100);
258  if(status == WAIT_OBJECT_0){
259  if(counter % 2 == 0 && !is_nt4){
260  /*
261  * Do nothing, since
262  * "If a change occurs after a call
263  * to FindFirstChangeNotification but
264  * before a call to FindNextChangeNotification,
265  * the operating system records the change.
266  * When FindNextChangeNotification is executed,
267  * the recorded change immediately satisfies
268  * a wait for the change notification." (MSDN)
269  * And so on... it happens between
270  * FindNextChangeNotification calls too.
271  *
272  * However, everything mentioned above is not true
273  * when the program modifies the file itself.
274  */
275  } else {
276  /* synchronize preferences reload with map redraw */
277  if(WaitForSingleObject(hMapEvent,INFINITE) != WAIT_OBJECT_0){
278  WgxDbgPrintLastError("PrefsChangesTrackingProc: wait on hMapEvent failed");
279  } else {
280  /* save state */
281  memcpy(&rc,&r_rc,sizeof(RECT));
282  s_maximized = maximized_window;
283  s_init_maximized = init_maximized_window;
284  s_skip_removable = skip_removable;
285  s_repeat_action = repeat_action;
286  memcpy(&cw,&user_defined_column_widths,sizeof(user_defined_column_widths));
287  s_list_height = list_height;
288  s_job_flags = job_flags;
289  s_show_taskbar_icon_overlay = show_taskbar_icon_overlay;
290 
291  /* reload preferences */
292  GetPrefs();
293 
294  /* restore state */
295  memcpy(&r_rc,&rc,sizeof(RECT));
296  maximized_window = s_maximized;
297  init_maximized_window = s_init_maximized;
298  skip_removable = s_skip_removable;
299  repeat_action = s_repeat_action;
300  memcpy(&user_defined_column_widths,&cw,sizeof(user_defined_column_widths));
301  list_height = s_list_height;
302  job_flags = s_job_flags;
303 
304  SetEvent(hMapEvent);
305 
306  if(dry_run == 0){
307  if(portable_mode) SetWindowText(hWindow,VERSIONINTITLE_PORTABLE);
308  else SetWindowText(hWindow,VERSIONINTITLE);
309  } else {
310  if(portable_mode) SetWindowText(hWindow,VERSIONINTITLE_PORTABLE " (dry run)");
311  else SetWindowText(hWindow,VERSIONINTITLE " (dry run)");
312  }
313 
314  /* if block size or grid line width changed since last redraw, resize map */
315  if(map_block_size != last_block_size || grid_line_width != last_grid_width){
316  /* ResizeMap is not safe to be called from threads */
317  map_rc.left = last_x;
318  map_rc.top = last_y;
319  map_rc.right = last_width;
320  map_rc.bottom = last_height;
321  PostMessage(hMap,WM_RESIZE_MAP,0,(LPARAM)&map_rc);
322  InvalidateRect(hMap,NULL,TRUE);
323  UpdateWindow(hMap);
324  } else {
325  /* redraw map if grid color changed */
326  RedrawMap(current_job,0);
327  }
328 
329  /* handle show_taskbar_icon_overlay option adjustment */
330  if(WaitForSingleObject(hTaskbarIconEvent,INFINITE) != WAIT_OBJECT_0){
331  WgxDbgPrintLastError("PrefsChangesTrackingProc: wait on hTaskbarIconEvent failed");
332  } else {
333  if(show_taskbar_icon_overlay != s_show_taskbar_icon_overlay){
334  if(show_taskbar_icon_overlay && job_is_running)
335  SetTaskbarIconOverlay(IDI_BUSY,"JOB_IS_RUNNING");
336  else
337  RemoveTaskbarIconOverlay();
338  }
339  SetEvent(hTaskbarIconEvent);
340  }
341  }
342  }
343  counter ++;
344  /* wait for the next notification */
345  if(!FindNextChangeNotification(h)){
346  WgxDbgPrintLastError("PrefsChangesTrackingProc: FindNextChangeNotification failed");
347  break;
348  }
349  }
350  }
351 
352  /* cleanup */
353  FindCloseChangeNotification(h);
354  changes_tracking_stopped = 1;
355  return 0;
356 }

385 int IsBootTimeDefragEnabled(void)
386 {
387  HKEY hKey;
388  DWORD type, size;
389  char *data, *curr_pos;
390  DWORD i, length, curr_len;
391 
392  if(RegOpenKeyEx(HKEY_LOCAL_MACHINE,
393  "SYSTEM\\CurrentControlSet\\Control\\Session Manager",
394  0,
395  KEY_QUERY_VALUE,
396  &hKey) != ERROR_SUCCESS){
397  WgxDisplayLastError(hWindow,MB_OK | MB_ICONHAND,"Cannot open SMSS key!");
398  return FALSE;
399  }
400 
401  type = REG_MULTI_SZ;
402  (void)RegQueryValueEx(hKey,"BootExecute",NULL,&type,NULL,&size);
403  data = malloc(size + 10);
404  if(!data){
405  (void)RegCloseKey(hKey);
406  MessageBox(0,"Not enough memory for IsBootTimeDefragEnabled()!",
407  "Error",MB_OK | MB_ICONHAND);
408  return FALSE;
409  }
410 
411  type = REG_MULTI_SZ;
412  if(RegQueryValueEx(hKey,"BootExecute",NULL,&type,
413  data,&size) != ERROR_SUCCESS){
414  WgxDisplayLastError(hWindow,MB_OK | MB_ICONHAND,"Cannot query BootExecute value!");
415  (void)RegCloseKey(hKey);
416  free(data);
417  return FALSE;
418  }
419 
420  length = size - 1;
421  for(i = 0; i < length;){
422  curr_pos = data + i;
423  curr_len = strlen(curr_pos) + 1;
424  /* if the command is yet registered then exit */
425  if(!strcmp(curr_pos,"defrag_native")){
426  (void)RegCloseKey(hKey);
427  free(data);
428  return TRUE;
429  }
430  i += curr_len;
431  }
432 
433  (void)RegCloseKey(hKey);
434  free(data);
435  return FALSE;
436 }

442 DWORD WINAPI BootExecTrackingProc(LPVOID lpParameter)
443 {
444  HKEY hKey;
445  HANDLE hEvent;
446  LONG error;
447 
448  if(RegOpenKeyEx(HKEY_LOCAL_MACHINE,
449  "SYSTEM\\CurrentControlSet\\Control\\Session Manager",
450  0,
451  KEY_NOTIFY,
452  &hKey) != ERROR_SUCCESS){
453  WgxDbgPrintLastError("BootExecTrackingProc: cannot open SMSS key");
454  boot_exec_tracking_stopped = 1;
455  return 0;
456  }
457 
458  hEvent = CreateEvent(NULL,FALSE,FALSE,NULL);
459  if(hEvent == NULL){
460  WgxDbgPrintLastError("BootExecTrackingProc: CreateEvent failed");
461  goto done;
462  }
463 
464 track_again:
465  error = RegNotifyChangeKeyValue(hKey,FALSE,
466  REG_NOTIFY_CHANGE_LAST_SET,hEvent,TRUE);
467  if(error != ERROR_SUCCESS){
468  WgxDbgPrint("BootExecTrackingProc: RegNotifyChangeKeyValue failed with code 0x%x",(UINT)error);
469  CloseHandle(hEvent);
470  goto done;
471  }
472 
473  while(!stop_track_boot_exec){
474  if(WaitForSingleObject(hEvent,100) == WAIT_OBJECT_0){
475  if(IsBootTimeDefragEnabled()){
476  WgxDbgPrint("Boot time defragmenter enabled (externally)\n");
477  boot_time_defrag_enabled = 1;
478  CheckMenuItem(hMainMenu,
479  IDM_CFG_BOOT_ENABLE,
480  MF_BYCOMMAND | MF_CHECKED);
481  SendMessage(hToolbar,TB_CHECKBUTTON,IDM_CFG_BOOT_ENABLE,MAKELONG(TRUE,0));
482  } else {
483  WgxDbgPrint("Boot time defragmenter disabled (externally)\n");
484  boot_time_defrag_enabled = 0;
485  CheckMenuItem(hMainMenu,
486  IDM_CFG_BOOT_ENABLE,
487  MF_BYCOMMAND | MF_UNCHECKED);
488  SendMessage(hToolbar,TB_CHECKBUTTON,IDM_CFG_BOOT_ENABLE,MAKELONG(FALSE,0));
489  }
490  goto track_again;
491  }
492  }
493  CloseHandle(hEvent);
494 
495 done:
496  RegCloseKey(hKey);
497  boot_exec_tracking_stopped = 1;
498  return 0;
499 }
