153 static int IsPortable(void)
154 {
155  wchar_t cd[MAX_PATH];
156  wchar_t instdir[MAX_PATH];
157  wchar_t path[MAX_PATH];
158  HKEY hRegKey = NULL;
159  DWORD path_length = (MAX_PATH - 1) * sizeof(wchar_t);
160  REGSAM samDesired = KEY_READ;
161  OSVERSIONINFO osvi;
162  BOOL bIsWindowsXPorLater;
163  LONG result;
164  int i;
165 
166  if(!GetModuleFileNameW(NULL,cd,MAX_PATH)){
167  WgxDbgPrintLastError("IsPortable: cannot get module file name");
168  return 1;
169  }
170 
171  /* make sure we have a terminating null character,
172  if the path is truncated on WinXP and earlier */
173  cd[MAX_PATH-1] = 0;
174 
175  /* strip off the file name */
176  i = wcslen(cd);
177  while(i >= 0) {
178  if(cd[i] == '\\'){
179  cd[i] = 0;
180  break;
181  }
182  i--;
183  }
184 
185  /* only XP and later support 32-bit registry view, so check for it */
186  ZeroMemory(&osvi, sizeof(OSVERSIONINFO));
187  osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
188 
189  GetVersionEx(&osvi);
190 
191  bIsWindowsXPorLater =
192  ( (osvi.dwMajorVersion > 5) ||
193  ( (osvi.dwMajorVersion == 5) && (osvi.dwMinorVersion >= 1) ));
194 
195  if(bIsWindowsXPorLater)
196  samDesired |= KEY_WOW64_32KEY;
197 
198  /* get install location from uninstall registry key */
199  result = RegOpenKeyEx(HKEY_LOCAL_MACHINE,
200  "Software\\Microsoft\\Windows\\CurrentVersion\\Uninstall\\UltraDefrag",
201  0,samDesired,&hRegKey);
202  if(result != ERROR_SUCCESS){
203  SetLastError((DWORD)result);
204  WgxDbgPrintLastError("IsPortable: cannot open registry");
205  return 1;
206  }
207 
208  result = RegQueryValueExW(hRegKey,L"InstallLocation",NULL,NULL,(LPBYTE) &instdir,&path_length);
209  RegCloseKey(hRegKey);
210  if(result != ERROR_SUCCESS){
211  SetLastError((DWORD)result);
212  WgxDbgPrintLastError("IsPortable: cannot read registry");
213  return 1;
214  }
215 
216  /* make sure we have a trailing zero character */
217  instdir[path_length / sizeof(wchar_t)] = 0;
218 
219  /* strip off any double quotes */
220  if(instdir[0] == '"'){
221  (void)wcsncpy(path,&instdir[1],wcslen(instdir)-2);
222  path[wcslen(instdir)-2] = 0;
223  } else {
224  (void)wcscpy(path,instdir);
225  }
226 
227  if(udefrag_wcsicmp(path,cd) == 0){
228  WgxDbgPrint("Install location \"%ws\" matches \"%ws\", so it isn't portable\n",path,cd);
229  return 0;
230  }
231 
232  WgxDbgPrint("Install location \"%ws\" differs from \"%ws\", so it is portable\n",path,cd);
233  return 1;
234 }

350 void ResizeMainWindow(int force)
351 {
352  int vlist_height, sbar_height;
353  int spacing;
354  int toolbar_height;
355  RECT rc;
356  int w, h;
357  int min_list_height;
358  int max_list_height;
359  int expand_list = 0;
360 
361  if(hStatus == NULL || hMainMenu == NULL || hToolbar == NULL)
362  return; /* this happens on early stages of window initialization */
363 
364  if(!GetClientRect(hWindow,&rc)){
365  WgxDbgPrint("GetClientRect failed in RepositionMainWindowControls()!\n");
366  return;
367  }
368 
369  if(list_height == 0)
370  list_height = DPI(VLIST_HEIGHT);
371 
372  /* correct invalid list heights */
373  min_list_height = GetMinVolListHeight();
374  if(list_height < min_list_height)
375  list_height = min_list_height;
376 
377  max_list_height = GetMaxVolListHeight();
378  if(list_height > max_list_height)
379  list_height = max_list_height;
380 
381  vlist_height = list_height;//DPI(VLIST_HEIGHT);
382  spacing = DPI(SPACING);
383  w = rc.right - rc.left;
384  h = rc.bottom - rc.top;
385 
386  /* prevent resizing if the current size is equal to previous */
387  if(!force && (prev_rc.right - prev_rc.left == w) && (prev_rc.bottom - prev_rc.top == h))
388  return; /* this usually encounters when user minimizes window and then restores it */
389  memcpy((void *)&prev_rc,(void *)&rc,sizeof(RECT));
390 
391  if(GetWindowRect(hToolbar,&rc))
392  toolbar_height = rc.bottom - rc.top;
393  else
394  toolbar_height = 24 + 2 * GetSystemMetrics(SM_CYEDGE);
395 
396  if(vlist_height > previous_list_height){
397  expand_list = 1;
398  previous_list_height = vlist_height;
399  }
400 
401  list_height = vlist_height = ResizeVolList(0,toolbar_height,w,vlist_height,expand_list);
402  previous_list_height = vlist_height;
403  /* do it twice for a proper list redraw when maximized window gets restored */
404  list_height = vlist_height = ResizeVolList(0,toolbar_height,w,vlist_height,0);
405  previous_list_height = vlist_height;
406  sbar_height = ResizeStatusBar(h,w);
407  ResizeMap(0,toolbar_height + vlist_height + 1,w,h - toolbar_height - vlist_height - sbar_height);
408 
409  /* redraw menu bar */
410  DrawMenuBar(hWindow);
411 
412  /* redraw toolbar */
413  SendMessage(hToolbar,TB_AUTOSIZE,0,0);
414 }
