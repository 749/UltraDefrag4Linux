139 static void deliver_message(char *string)
140 {
141  HANDLE hEvtBufferReady = NULL;
142  HANDLE hEvtDataReady = NULL;
143  HANDLE hSection = NULL;
144  LPVOID BaseAddress = NULL;
145  LARGE_INTEGER SectionOffset;
146  ULONG ViewSize = 0;
147  UNICODE_STRING us;
148  OBJECT_ATTRIBUTES oa;
149  NTSTATUS Status;
150  LARGE_INTEGER interval;
151  DBG_OUTPUT_DEBUG_STRING_BUFFER *dbuffer;
152  int length;
153 
154  /* open debugger's objects */
155  RtlInitUnicodeString(&us,L"\\BaseNamedObjects\\DBWIN_BUFFER_READY");
156  InitializeObjectAttributes(&oa,&us,0,NULL,NULL);
157  Status = NtOpenEvent(&hEvtBufferReady,SYNCHRONIZE,&oa);
158  if(!NT_SUCCESS(Status)) goto done;
159 
160  RtlInitUnicodeString(&us,L"\\BaseNamedObjects\\DBWIN_DATA_READY");
161  InitializeObjectAttributes(&oa,&us,0,NULL,NULL);
162  Status = NtOpenEvent(&hEvtDataReady,EVENT_MODIFY_STATE,&oa);
163  if(!NT_SUCCESS(Status)) goto done;
164 
165  RtlInitUnicodeString(&us,L"\\BaseNamedObjects\\DBWIN_BUFFER");
166  InitializeObjectAttributes(&oa,&us,0,NULL,NULL);
167  Status = NtOpenSection(&hSection,SECTION_ALL_ACCESS,&oa);
168  if(!NT_SUCCESS(Status)) goto done;
169  SectionOffset.QuadPart = 0;
170  Status = NtMapViewOfSection(hSection,NtCurrentProcess(),
171  &BaseAddress,0,0,&SectionOffset,(SIZE_T *)&ViewSize,ViewShare,
172  0,PAGE_READWRITE);
173  if(!NT_SUCCESS(Status)) goto done;
174 
175  /* send a message */
176  /*
177  * wait a maximum of 10 seconds for the debug monitor
178  * to finish processing the shared buffer
179  */
180  interval.QuadPart = -(10000 * 10000);
181  if(NtWaitForSingleObject(hEvtBufferReady,FALSE,&interval) != WAIT_OBJECT_0)
182  goto done;
183 
184  /* write the process id into the buffer */
185  dbuffer = (DBG_OUTPUT_DEBUG_STRING_BUFFER *)BaseAddress;
186  dbuffer->ProcessId = (DWORD)(DWORD_PTR)(NtCurrentTeb()->ClientId.UniqueProcess);
187 
188  (void)strncpy(dbuffer->Msg,string,DBG_OUT_BUFFER_SIZE);
189  dbuffer->Msg[DBG_OUT_BUFFER_SIZE - 1] = 0;
190 
191  /* ensure that the buffer has new line character */
192  length = strlen(dbuffer->Msg);
193  if(length > 0){
194  if(dbuffer->Msg[length-1] != '\n'){
195  if(length == (DBG_OUT_BUFFER_SIZE - 1)){
196  dbuffer->Msg[length-1] = '\n';
197  } else {
198  dbuffer->Msg[length] = '\n';
199  dbuffer->Msg[length+1] = 0;
200  }
201  }
202  } else {
203  strcpy(dbuffer->Msg,"\n");
204  }
205 
206  /* signal that the buffer contains meaningful data and can be read */
207  (void)NtSetEvent(hEvtDataReady,NULL);
208 
209 done:
210  NtCloseSafe(hEvtBufferReady);
211  NtCloseSafe(hEvtDataReady);
212  if(BaseAddress)
213  (void)NtUnmapViewOfSection(NtCurrentProcess(),BaseAddress);
214  NtCloseSafe(hSection);
215 }

236 static void *get_description(ULONG error,int *encoding)
237 {
238  UNICODE_STRING uStr;
239  NTSTATUS Status;
240  HMODULE base_addr;
241  MESSAGE_RESOURCE_ENTRY *mre;
242 
243  /* kernel32.dll has much better messages than ntdll.dll */
244  RtlInitUnicodeString(&uStr,L"kernel32.dll");
245  Status = LdrGetDllHandle(0,0,&uStr,&base_addr);
246  if(!NT_SUCCESS(Status)){
247  RtlInitUnicodeString(&uStr,L"ntdll.dll");
248  Status = LdrGetDllHandle(0,0,&uStr,&base_addr);
249  if(!NT_SUCCESS(Status))
250  return NULL; /* this case is very extraordinary */
251  }
252  Status = RtlFindMessage(base_addr,(ULONG)(DWORD_PTR)RT_MESSAGETABLE,
253  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),error,&mre);
254  if(!NT_SUCCESS(Status))
255  return NULL; /* no appropriate message found */
256  if(encoding){
257  *encoding = (mre->Flags & MESSAGE_RESOURCE_UNICODE) ? ENC_UTF16 : ENC_ANSI;
258  }
259  return (void *)(mre->Text);
260 }

276 void winx_dbg_print(char *format, ...)
277 {
278  char *p, *msg = NULL;
279  char *err_msg = NULL;
280  char *ext_msg = NULL;
281  char *cnv_msg = NULL;
282  ULONG status, error;
283  int ns_flag = 0, le_flag = 0;
284  int encoding;
285  int length;
286  va_list arg;
287 
288  /* save last error codes */
289  status = NtCurrentTeb()->LastStatusValue;
290  error = NtCurrentTeb()->LastErrorValue;
291 
292  /* format message */
293  if(format){
294  va_start(arg,format);
295  msg = winx_vsprintf(format,arg);
296  va_end(arg);
297  }
298  if(!msg) return;
299 
300  /* get rid of trailing new line character */
301  length = strlen(msg);
302  if(length){
303  if(msg[length - 1] == '\n')
304  msg[length - 1] = 0;
305  }
306 
307  /* check for $NS and $LE magic sequences */
308  p = strstr(msg,": $NS");
309  if(p == msg + length - 5){
310  msg[length - 5] = 0;
311  error = RtlNtStatusToDosError(status);
312  ns_flag = 1;
313  } else {
314  p = strstr(msg,": $LE");
315  if(p == msg + length - 5){
316  msg[length - 5] = 0;
317  le_flag = 1;
318  }
319  }
320 
321  if(ns_flag || le_flag){
322  err_msg = get_description(error,&encoding);
323  if(err_msg){
324  if(encoding == ENC_ANSI){
325  ext_msg = winx_sprintf("%s: 0x%x %s: %s",
326  msg,ns_flag ? (UINT)status : (UINT)error,
327  ns_flag ? "status" : "error",err_msg);
328  add_dbg_log_entry(ext_msg ? ext_msg : msg);
329  deliver_message(ext_msg ? ext_msg : msg);
330  } else {
331  length = (wcslen((wchar_t *)err_msg) + 1) * sizeof(wchar_t);
332  length *= 2; /* enough to hold UTF-8 string */
333  cnv_msg = winx_heap_alloc(length);
334  if(cnv_msg){
335  /* write message to log in UTF-8 encoding */
336  winx_to_utf8(cnv_msg,length,(wchar_t *)err_msg);
337  ext_msg = winx_sprintf("%s: 0x%x %s: %s",
338  msg,ns_flag ? (UINT)status : (UINT)error,
339  ns_flag ? "status" : "error",cnv_msg);
340  add_dbg_log_entry(ext_msg ? ext_msg : msg);
341  winx_heap_free(ext_msg); ext_msg = NULL;
342  /* send message to debugger in ANSI encoding */
343  _snprintf(cnv_msg,length,"%ls",(wchar_t *)err_msg);
344  cnv_msg[length - 1] = 0;
345  ext_msg = winx_sprintf("%s: 0x%x %s: %s",
346  msg,ns_flag ? (UINT)status : (UINT)error,
347  ns_flag ? "status" : "error",cnv_msg);
348  deliver_message(ext_msg ? ext_msg : msg);
349  } else {
350  goto no_description;
351  }
352  }
353  } else {
354 no_description:
355  ext_msg = winx_sprintf("%s: 0x%x %s",
356  msg,ns_flag ? (UINT)status : (UINT)error,
357  ns_flag ? "status" : "error");
358  add_dbg_log_entry(ext_msg ? ext_msg : msg);
359  deliver_message(ext_msg ? ext_msg : msg);
360  }
361  } else {
362  add_dbg_log_entry(msg);
363  deliver_message(msg);
364  }
365 
366  /* cleanup */
367  winx_heap_free(msg);
368  winx_heap_free(ext_msg);
369  winx_heap_free(cnv_msg);
370 }

563 static void flush_dbg_log(int already_synchronized)
564 {
565  #define DBG_BUFFER_SIZE (100 * 1024) /* 100 KB */
566  WINX_FILE *f;
567  char *lb;
568  winx_dbg_log_entry *old_dbg_log, *log_entry;
569  int length;
570  char crlf[] = "\r\n";
571  char *time_stamp;
572 
573  /* synchronize with other threads */
574  if(!already_synchronized){
575  if(winx_acquire_spin_lock(path_lock,INFINITE) < 0){
576  DebugPrint("flush_dbg_log: synchronization failed");
577  winx_print("\nflush_dbg_log: synchronization failed!\n");
578  return;
579  }
580  }
581 
582  /* disable parallel access to dbg_log list */
583  if(winx_acquire_spin_lock(dbg_lock,INFINITE) < 0){
584  if(!already_synchronized)
585  winx_release_spin_lock(path_lock);
586  return;
587  }
588  old_dbg_log = dbg_log;
589  dbg_log = NULL;
590  winx_release_spin_lock(dbg_lock);
591 
592  if(!old_dbg_log || !log_path)
593  goto done;
594 
595  if(log_path[0] == 0)
596  goto done;
597 
598  /* open log file */
599  f = winx_fbopen(log_path,"a",DBG_BUFFER_SIZE);
600  if(f == NULL){
601  /* recreate path if it does not exist */
602  lb = strrchr(log_path,'\\');
603  if(lb) *lb = 0;
604  if(winx_create_path(log_path) < 0){
605  DebugPrint("flush_old_dbg_log: cannot create directory tree for log path");
606  winx_print("\nflush_old_dbg_log: cannot create directory tree for log path\n");
607  }
608  if(lb) *lb = '\\';
609  f = winx_fbopen(log_path,"a",DBG_BUFFER_SIZE);
610  }
611 
612  /* save log */
613  if(f != NULL){
614  winx_printf("\nWriting log file \"%s\" ...\n",&log_path[4]);
615  for(log_entry = old_dbg_log; log_entry; log_entry = log_entry->next){
616  if(log_entry->buffer){
617  length = strlen(log_entry->buffer);
618  if(length){
619  /* get rid of trailing new line character */
620  if(log_entry->buffer[length - 1] == '\n'){
621  log_entry->buffer[length - 1] = 0;
622  length --;
623  }
624  time_stamp = winx_sprintf("%04d-%02d-%02d %02d:%02d:%02d.%03d ",
625  log_entry->time_stamp.year, log_entry->time_stamp.month, log_entry->time_stamp.day,
626  log_entry->time_stamp.hour, log_entry->time_stamp.minute, log_entry->time_stamp.second,
627  log_entry->time_stamp.milliseconds);
628  if(time_stamp){
629  (void)winx_fwrite(time_stamp,sizeof(char),strlen(time_stamp),f);
630  winx_heap_free(time_stamp);
631  }
632  (void)winx_fwrite(log_entry->buffer,sizeof(char),length,f);
633  /* add a proper newline characters */
634  (void)winx_fwrite(crlf,sizeof(char),2,f);
635  }
636  }
637  if(log_entry->next == old_dbg_log) break;
638  }
639  winx_fclose(f);
640  winx_list_destroy((list_entry **)(void *)&old_dbg_log);
641  }
642 
643 done:
644  /* end of synchronization */
645  if(!already_synchronized)
646  winx_release_spin_lock(path_lock);
647 }

698  DebugPrint("winx_enable_dbg_log: cannot allocate memory for log path");
699  winx_print("\nCannot allocate memory for log path!\n");
700  }
701  }

703  /* end of synchronization */
704  winx_release_spin_lock(path_lock);
