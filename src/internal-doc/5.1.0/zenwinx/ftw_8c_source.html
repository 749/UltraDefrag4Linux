132 int winx_ftw_dump_file(winx_file_info *f,
133  ftw_terminator t, void *user_defined_data)
134 {
135  GET_RETRIEVAL_DESCRIPTOR *filemap;
136  HANDLE hFile;
137  ULONGLONG startVcn;
138  long counter; /* counts number of attempts to get information */
139  #define MAX_COUNT 1000
140  IO_STATUS_BLOCK iosb;
141  NTSTATUS status;
142  int i;
143  winx_blockmap *block = NULL;
144 
145  DbgCheck1(f,"winx_ftw_dump_file",-1);
146 
147  /* reset disposition related fields */
148  f->disp.clusters = 0;
149  f->disp.fragments = 0;
150  f->disp.flags = 0;
151  winx_list_destroy((list_entry **)(void *)&f->disp.blockmap);
152 
153  /* open the file */
154  status = winx_defrag_fopen(f,WINX_OPEN_FOR_DUMP,&hFile);
155  if(status != STATUS_SUCCESS){
156  DebugPrintEx(status,"winx_ftw_dump_file: cannot open %ws",f->path);
157  return 0; /* file is locked by system */
158  }
159 
160  /* allocate memory */
161  filemap = winx_heap_alloc(FILE_MAP_SIZE);
162  if(filemap == NULL){
163  DebugPrint("winx_ftw_dump_file: cannot allocate %u bytes of memory",
164  FILE_MAP_SIZE);
165  winx_defrag_fclose(hFile);
166  return (-1);
167  }
168 
169  /* dump the file */
170  startVcn = 0;
171  counter = 0;
172  do {
173  memset(filemap,0,FILE_MAP_SIZE);
174  status = NtFsControlFile(hFile,NULL,NULL,0,
175  &iosb,FSCTL_GET_RETRIEVAL_POINTERS,
176  &startVcn,sizeof(ULONGLONG),
177  filemap,FILE_MAP_SIZE);
178  counter ++;
179  if(NT_SUCCESS(status)){
180  NtWaitForSingleObject(hFile,FALSE,NULL);
181  status = iosb.Status;
182  }
183  if(status != STATUS_SUCCESS && status != STATUS_BUFFER_OVERFLOW){
184  /* it always returns STATUS_END_OF_FILE for small files placed inside MFT */
185  if(status == STATUS_END_OF_FILE) goto empty_map_detected;
186  DebugPrintEx(status,"winx_ftw_dump_file: dump failed for %ws",f->path);
187  goto dump_failed;
188  }
189 
190  if(ftw_check_for_termination(t,user_defined_data)){
191  if(counter > MAX_COUNT)
192  DebugPrint("winx_ftw_dump_file: %ws: infinite main loop?",f->path);
193  /* reset incomplete map */
194  goto cleanup;
195  }
196 
197  /* check for an empty map */
198  if(!filemap->NumberOfPairs && status != STATUS_SUCCESS){
199  DebugPrint("winx_ftw_dump_file: %ws: empty map of file detected",f->path);
200  goto empty_map_detected;
201  }
202 
203  /* loop through the buffer of number/cluster pairs */
204  startVcn = filemap->StartVcn;
205  for(i = 0; i < (ULONGLONG)filemap->NumberOfPairs; startVcn = filemap->Pair[i].Vcn, i++){
206  /* compressed virtual runs (0-filled) are identified with LCN == -1 */
207  if(filemap->Pair[i].Lcn == LLINVALID)
208  continue;
209 
210  /* the following is usual for 3.99 GB files on FAT32 under XP */
211  if(filemap->Pair[i].Vcn == 0){
212  DebugPrint("winx_ftw_dump_file: %ws: wrong map of file detected",f->path);
213  goto dump_failed;
214  }
215 
216  block = (winx_blockmap *)winx_list_insert_item((list_entry **)&f->disp.blockmap,
217  (list_entry *)block,sizeof(winx_blockmap));
218  if(block == NULL){
219  DebugPrint("winx_ftw_dump_file: cannot allocate %u bytes of memory",
220  sizeof(winx_blockmap));
221  goto dump_failed;
222  }
223  block->lcn = filemap->Pair[i].Lcn;
224  block->length = filemap->Pair[i].Vcn - startVcn;
225  block->vcn = startVcn;
226 
227  //DebugPrint("VCN = %I64u, LCN = %I64u, LENGTH = %I64u",
228  // block->vcn,block->lcn,block->length);
229 
230  f->disp.clusters += block->length;
231  if(block == f->disp.blockmap)
232  f->disp.fragments ++;
233 
234  /*
235  * Sometimes files have more than one fragment,
236  * but are not fragmented yet. In case of compressed
237  * files this happens quite frequently.
238  */
239  if(block != f->disp.blockmap && \
240  block->lcn != (block->prev->lcn + block->prev->length)){
241  f->disp.fragments ++;
242  f->disp.flags |= WINX_FILE_DISP_FRAGMENTED;
243  }
244  }
245  } while(status != STATUS_SUCCESS);
246  /* small directories placed inside MFT have empty list of fragments... */
247 
248  /* the dump is completed */
249  validate_blockmap(f);
250  winx_heap_free(filemap);
251  winx_defrag_fclose(hFile);
252  return 0;
253 
254 cleanup:
255 empty_map_detected:
256  f->disp.clusters = 0;
257  f->disp.fragments = 0;
258  f->disp.flags = 0;
259  winx_list_destroy((list_entry **)(void *)&f->disp.blockmap);
260  winx_heap_free(filemap);
261  winx_defrag_fclose(hFile);
262  return 0;
263 
264 dump_failed:
265  f->disp.clusters = 0;
266  f->disp.fragments = 0;
267  f->disp.flags = 0;
268  winx_list_destroy((list_entry **)(void *)&f->disp.blockmap);
269  winx_heap_free(filemap);
270  winx_defrag_fclose(hFile);
271  return (-1);
272 }

333  return NULL;
334  }

336  (void)_snwprintf(f->path,length,L"%ws%ws",path,f->name);
337  else
338  (void)_snwprintf(f->path,length,L"%ws\\%ws",path,f->name);
339  f->path[length - 1] = 0;
340 
341  /* save file attributes and access times */
342  f->flags = file_entry->FileAttributes;
343  f->creation_time = file_entry->CreationTime.QuadPart;
344  f->last_modification_time = file_entry->LastWriteTime.QuadPart;
345  f->last_access_time = file_entry->LastAccessTime.QuadPart;
346 
347  /* reset user defined flags */
348  f->user_defined_flags = 0;
349 
350  //DebugPrint("%ws",f->path);
351 
352  /* reset internal data fields */
353  memset(&f->internal,0,sizeof(winx_file_internal_info));
354 
355  /* reset file disposition */
356  memset(&f->disp,0,sizeof(winx_file_disposition));
357 
358  /* get file disposition if requested */
359  if(flags & WINX_FTW_DUMP_FILES){
360  if(winx_ftw_dump_file(f,t,user_defined_data) < 0){
361  winx_heap_free(f->name);
362  winx_heap_free(f->path);
363  winx_list_remove_item((list_entry **)(void *)filelist,(list_entry *)f);
364  return NULL;
365  }
366  }
367 
368  return f;
369 }

520 static int ftw_helper(short *path, int flags,
521  ftw_filter_callback fcb, ftw_progress_callback pcb,
522  ftw_terminator t, void *user_defined_data,
523  winx_file_info **filelist)
524 {
525  FILE_BOTH_DIR_INFORMATION *file_listing, *file_entry;
526  HANDLE hDir;
527  IO_STATUS_BLOCK iosb;
528  NTSTATUS status;
529  winx_file_info *f;
530  int skip_children, result;
531 
532  /* open directory */
533  hDir = ftw_open_directory(path);
534  if(hDir == NULL)
535  return 0; /* directory is locked by system, skip it */
536 
537  /* allocate memory */
538  file_listing = winx_heap_alloc(FILE_LISTING_SIZE);
539  if(file_listing == NULL){
540  DebugPrint("ftw_helper: cannot allocate %u bytes of memory",
541  FILE_LISTING_SIZE);
542  NtClose(hDir);
543  return (-1);
544  }
545 
546  /* reset buffer */
547  memset((void *)file_listing,0,FILE_LISTING_SIZE);
548  file_entry = file_listing;
549 
550  /* list directory entries */
551  while(!ftw_check_for_termination(t,user_defined_data)){
552  /* get directory entry */
553  if(file_entry->NextEntryOffset){
554  /* go to the next directory entry */
555  file_entry = (FILE_BOTH_DIR_INFORMATION *)((char *)file_entry + \
556  file_entry->NextEntryOffset);
557  } else {
558  /* read next portion of directory entries */
559  memset((void *)file_listing,0,FILE_LISTING_SIZE);
560  status = NtQueryDirectoryFile(hDir,NULL,NULL,NULL,
561  &iosb,(void *)file_listing,FILE_LISTING_SIZE,
562  FileBothDirectoryInformation,
563  FALSE /* return multiple entries */,
564  NULL,
565  FALSE /* do not restart scan */
566  );
567  if(status != STATUS_SUCCESS){
568  if(status != STATUS_NO_MORE_FILES)
569  DebugPrintEx(status,"ftw_helper failed");
570  /* no more entries to read */
571  winx_heap_free(file_listing);
572  NtClose(hDir);
573  return 0;
574  }
575  file_entry = file_listing;
576  }
577 
578  /* skip . and .. entries */
579  if(file_entry->FileNameLength == sizeof(short)){
580  if(file_entry->FileName[0] == '.')
581  continue;
582  }
583  if(file_entry->FileNameLength == 2 * sizeof(short)){
584  if(file_entry->FileName[0] == '.' && file_entry->FileName[1] == '.')
585  continue;
586  }
587 
588  /* validate entry */
589  if(file_entry->FileNameLength == 0)
590  continue;
591 
592  /* add entry to the file list */
593  f = ftw_add_entry_to_filelist(path,flags,fcb,pcb,t,
594  user_defined_data,filelist,file_entry);
595  if(f == NULL){
596  winx_heap_free(file_listing);
597  NtClose(hDir);
598  return (-1);
599  }
600 
601  //DebugPrint("%ws\n%ws",f->name,f->path);
602 
603  /* check for termination */
604  if(ftw_check_for_termination(t,user_defined_data)){
605  DebugPrint("ftw_helper: terminated by user");
606  winx_heap_free(file_listing);
607  NtClose(hDir);
608  return (-2);
609  }
610 
611  /* call the callback routines */
612  if(pcb != NULL)
613  pcb(f,user_defined_data);
614 
615  skip_children = 0;
616  if(fcb != NULL)
617  skip_children = fcb(f,user_defined_data);
618 
619  /* scan subdirectories if requested */
620  if(is_directory(f) && (flags & WINX_FTW_RECURSIVE) && !skip_children){
621  /* don't follow reparse points! */
622  if(!is_reparse_point(f)){
623  result = ftw_helper(f->path,flags,fcb,pcb,t,user_defined_data,filelist);
624  if(result < 0){
625  winx_heap_free(file_listing);
626  NtClose(hDir);
627  return result;
628  }
629  }
630  }
631  }
632 
633  /* terminated */
634  winx_heap_free(file_listing);
635  NtClose(hDir);
636  return (-2);
637 }

815 winx_file_info *winx_scan_disk(char volume_letter, int flags,
816  ftw_filter_callback fcb, ftw_progress_callback pcb, ftw_terminator t,
817  void *user_defined_data)
818 {
819  winx_file_info *filelist = NULL;
820  wchar_t rootpath[] = L"\\??\\A:\\";
821  winx_volume_information v;
822  ULONGLONG time;
823 
824  /* ensure that it will work on w2k */
825  volume_letter = winx_toupper(volume_letter);
826 
827  time = winx_xtime();
828  winx_dbg_print_header(0,0,"winx_scan_disk started");
829 
830  if(flags & WINX_FTW_SKIP_RESIDENT_STREAMS){
831  if(!(flags & WINX_FTW_DUMP_FILES)){
832  DebugPrint("winx_ftw: WINX_FTW_DUMP_FILES flag"
833  " must be set to accept WINX_FTW_SKIP_RESIDENT_STREAMS");
834  flags &= ~WINX_FTW_SKIP_RESIDENT_STREAMS;
835  }
836  }
837 
838  if(winx_get_volume_information(volume_letter,&v) >= 0){
839  DebugPrint("winx_scan_disk: file system is %s",v.fs_name);
840  if(!strcmp(v.fs_name,"NTFS")){
841  filelist = ntfs_scan_disk(volume_letter,flags,fcb,pcb,t,user_defined_data);
842  goto cleanup;
843  }
844  }
845 
846  /* collect information about root directory */
847  rootpath[4] = (short)volume_letter;
848  if(ftw_add_root_directory(rootpath,flags,fcb,pcb,t,user_defined_data,&filelist) == (-1) && \
849  !(flags & WINX_FTW_ALLOW_PARTIAL_SCAN)){
850  /* destroy list */
851  winx_ftw_release(filelist);
852  filelist = NULL;
853  goto done;
854  }
855 
856  /* collect information about entire directory tree */
857  flags |= WINX_FTW_RECURSIVE;
858  if(ftw_helper(rootpath,flags,fcb,pcb,t,user_defined_data,&filelist) == (-1) && \
859  !(flags & WINX_FTW_ALLOW_PARTIAL_SCAN)){
860  /* destroy list */
861  winx_ftw_release(filelist);
862  filelist = NULL;
863  goto done;
864  }
865 
866 cleanup:
867  if(flags & WINX_FTW_SKIP_RESIDENT_STREAMS)
868  ftw_remove_resident_streams(&filelist);
869  /* get rid of invalid entries */
870  ftw_remove_invalid_streams(&filelist);
871 
872 done:
873  winx_dbg_print_header(0,0,"winx_scan_disk completed in %I64u ms",
874  winx_xtime() - time);
875  return filelist;
876 }

885 void winx_ftw_release(winx_file_info *filelist)
886 {

888 
889  /* walk through list of files and free allocated memory */
890  for(f = filelist; f != NULL; f = f->next){

899 }
