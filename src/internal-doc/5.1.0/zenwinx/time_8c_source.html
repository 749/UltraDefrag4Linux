35 ULONGLONG winx_str2time(char *string)
36 {
37  ULONGLONG time = 0;
38  char buffer[128] = "";
39  int index = 0;
40  int i;
41  char c;
42  ULONGLONG k;
43 
44  if(string == NULL)
45  return 0;
46 
47  for(i = 0;; i++){ /* loop through all characters of the string */
48  c = string[i];
49  if(!c) break;
50  if(c >= '0' && c <= '9'){
51  buffer[index] = c;
52  index++;
53  }
54 
55  k = 0;
56  c = winx_toupper(c);
57  switch(c){
58  case 'S':
59  k = 1;
60  break;
61  case 'M':
62  k = 60;
63  break;
64  case 'H':
65  k = 3600;
66  break;
67  case 'D':
68  k = 3600 * 24;
69  break;
70  case 'Y':
71  k = 3600 * 24 * 356;
72  break;
73  }
74  if(k){
75  buffer[index] = 0;
76  index = 0;
77  time += k * _atoi64(buffer);
78  }
79  }
80  return time;

155  xtime = 1000 * counter.QuadPart;
156  if(xtime / 1000 != counter.QuadPart){
157  /* overflow occured; to avoid use of arbitrary
158  precision arithmetic let's round to seconds */
159  xtime = 1000 * (counter.QuadPart / frequency.QuadPart);
160  } else {
161  xtime /= frequency.QuadPart;
162  }
