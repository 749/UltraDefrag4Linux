259 static int wcsmatch_helper(wchar_t *string, wchar_t *mask)
260 {
261  wchar_t cs, cm;
262 
263  while(*string && *mask){
264  cs = *string; cm = *mask;
265  if(cs != cm && cm != '?'){
266  /* the current pair of characters differs */
267  if(cm != '*') return 0;
268  /* skip asterisks */
269  while(*mask == '*') mask ++;
270  if(*mask == 0) return 1;
271  /* compare rest of the string with rest of the mask */
272  cm = *mask;
273  if(cm == '?'){
274  /* the question mark matches any single character */
275  for(; *string; string++){
276  if(wcsmatch_helper(string, mask))
277  return 1;
278  }
279  } else {
280  /* skip part of the string which doesn't match for sure */
281  for(; *string; string++){
282  if(*string == cm){
283  if(wcsmatch_helper(string, mask))
284  return 1;
285  }
286  }
287  }
288  return 0;
289  }
290  /* let's compare next pair of characters */
291  string ++;
292  mask ++;
293  }
294 
295  while(*mask == '*') mask ++;
296  return (*string == 0 && *mask == 0) ? 1 : 0;
297 }

445 int winx_patcomp(winx_patlist *patterns,wchar_t *string,wchar_t *delim,int flags)
446 {
447  int pattern_detected;
448  int i, j, n;
449  wchar_t *s;
450 
451  if(patterns == NULL || string == NULL || delim == NULL)
452  return (-1);
453 
454  /* reset patterns structure */
455  patterns->flags = flags;
456  patterns->count = 0;
457  patterns->array = NULL;
458  patterns->string = NULL;
459 
460  if(string[0] == 0)
461  return 0; /* empty list of patterns */
462 
463  /* make a copy of the string */
464  s = winx_wcsdup(string);
465  if(s == NULL){
466  DebugPrint("winx_patcomp: cannot allocate %u bytes of memory",
467  (wcslen(string) + 1) * sizeof(wchar_t));
468  return (-1);
469  }
470 
471  /* replace all delimiters by zeros */
472  for(n = 0; s[n]; n++){
473  if(wcschr(delim,s[n]))
474  s[n] = 0;
475  }
476 
477  /* count all patterns */
478  pattern_detected = 0;
479  for(i = 0; i < n; i++){
480  if(s[i] != 0){
481  if(!pattern_detected){
482  patterns->count ++;
483  pattern_detected = 1;
484  }
485  } else {
486  pattern_detected = 0;
487  }
488  }
489 
490  /* build array of patterns */
491  patterns->array = winx_heap_alloc(patterns->count * sizeof(wchar_t *));
492  if(patterns->array == NULL){
493  DebugPrint("winx_patcomp: cannot allocate %u bytes of memory",
494  patterns->count * sizeof(wchar_t *));
495  winx_heap_free(s);
496  patterns->count = 0;
497  return (-1);
498  }
499  pattern_detected = 0;
500  for(i = j = 0; i < n; i++){
501  if(s[i] != 0){
502  if(!pattern_detected){
503  patterns->array[j] = s + i;
504  j ++;
505  pattern_detected = 1;
506  }
507  } else {
508  pattern_detected = 0;
509  }
510  }
511 
512  patterns->string = s;
513  return 0;
514 }

608 int winx_bytes_to_hr(ULONGLONG bytes, int digits, char *buffer, int length)
609 {
610  char *suffixes[] = { "B", "KB", "MB", "GB", "TB", "PB", "EB", "ZB", "YB" };
611  ULONGLONG n; /* integer part */
612  ULONGLONG m; /* multiplier */
613  ULONGLONG r; /* remaining part */
614  int i; /* index for the suffixes array */
615  double rd;
616  char spec[] = "%I64u.%00I64u %s";
617  int result;
618 
619  DbgCheck3(digits >= 0, buffer != NULL, length > 0, "winx_bytes_to_hr", -1);
620 
621  for(n = bytes, m = 1, i = 0; n >> 10; n >>= 10, m <<= 10, i++){}
622  r = bytes - n * m;
623 
624  /* Win DDK cannot convert ULONGLONG to double directly, */
625  /* but now it's safe to convert both r and m through LONGLONG */
626  rd = (double)(LONGLONG)r / (double)(LONGLONG)m;
627  if(rd >= 1) rd = 0.999999999999999;
628  rd *= pow(10, digits);
629  /* convertion to LONGLONG is needed for MinGW */
630  /* it is safe, but may lower a highest possible precision */
631  r = (ULONGLONG)(LONGLONG)rd;
632 
633  if(digits == 0){
634  result = _snprintf(buffer, length - 1, "%I64u %s", n, suffixes[i]);
635  } else {
636  spec[7] = '0' + digits / 10;
637  spec[8] = '0' + digits % 10;
638  result = _snprintf(buffer, length - 1, spec, n, r, suffixes[i]);
639  }
640  return result;
641 }

653 ULONGLONG winx_hr_to_bytes(char *string)
654 {
655  char *suffixes[] = { "KB", "MB", "GB", "TB", "PB", "EB", "ZB", "YB" };
656  int suffix_found = 0;
657  char *dp; /* dot position */
658  ULONGLONG n; /* integer part */
659  ULONGLONG m; /* multiplier */
660  ULONGLONG r = 0; /* remaining part */
661  int i; /* index for the suffixes array */
662  int z; /* number of zeros after the dot */
663  double rd;
664 
665  DbgCheck1(string != NULL, "winx_hr_to_bytes", 0);
666 
667  n = (ULONGLONG)_atoi64(string);
668 
669  for(i = 0, m = 1024; i < sizeof(suffixes) / sizeof(char *); i++, m <<= 10){
670  if(winx_stristr(string,suffixes[i])){
671  suffix_found = 1;
672  break;
673  }
674  }
675  if(suffix_found == 0){
676  return n;
677  }
678 
679  dp = strchr(string, '.');
680  if(dp != NULL){
681  for(z = 0; dp[z + 1] == '0'; z++) {}
682  for(rd = (double)_atoi64(dp + 1); rd > 1; rd /= 10){}
683  /* convertion to LONGLONG is needed for MinGW */
684  r = (ULONGLONG)(LONGLONG)((double)(LONGLONG)m * rd * pow(10, -z));
685  }
686 
687  return n * m + r;
688 }

699 void winx_to_utf8(char *dst,int size,wchar_t *src)
700 {
701  int i; /* src index */
702  int j; /* dst index */
703  wchar_t c, b1, b2, b3;
704 
705  if(!src || !dst || size <= 0) return;
706 
707  for(i = j = 0; src[i]; i++){
708  c = src[i];
709  if(c < 0x80){
710  if(j > (size - 2)) break;
711  dst[j] = (char)c;
712  j ++;
713  } else if(c < 0x800){ /* 0x80 - 0x7FF: 2 bytes */
714  if(j > (size - 3)) break;
715  b2 = 0x80 | (c & 0x3F);
716  c >>= 6;
717  b1 = 0xC0 | c;
718  dst[j] = (char)b1;
719  dst[j+1] = (char)b2;
720  j += 2;
721  } else { /* 0x800 - 0xFFFF: 3 bytes */
722  if(j > (size - 4)) break;
723  b3 = 0x80 | (c & 0x3F);
724  c >>= 6;
725  b2 = 0x80 | (c & 0x3F);
726  c >>= 6;
727  b1 = 0xE0 | c;
728  dst[j] = (char)b1;
729  dst[j+1] = (char)b2;
730  dst[j+2] = (char)b3;
731  j += 3;
732  }
733  }
734  dst[j] = 0;
735 }
