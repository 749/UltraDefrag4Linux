35 void winx_sleep(int msec)
36 {
37  LARGE_INTEGER Interval;
38 
39  if(msec != INFINITE){
40  /* System time units are 100 nanoseconds. */
41  Interval.QuadPart = -((signed long)msec * 10000);
42  } else {
43  /* Approximately 292000 years hence */
44  Interval.QuadPart = MAX_WAIT_INTERVAL;
45  }
46  /* don't litter debugging log in case of errors */
47  (void)NtDelayExecution(0/*FALSE*/,&Interval);
48 }

65 int winx_get_os_version(void)

241 short *winx_get_windows_boot_options(void)
242 {
243  UNICODE_STRING us;
244  OBJECT_ATTRIBUTES oa;
245  NTSTATUS status;
246  HANDLE hKey;
247  KEY_VALUE_PARTIAL_INFORMATION *data;
248  short *data_buffer = NULL;
249  DWORD data_size = 0;
250  DWORD data_size2 = 0;
251  DWORD data_length;
252  BOOLEAN empty_value = FALSE;
253  short *boot_options;
254  int buffer_size;
255 
256  /* 1. open HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control registry key */
257  RtlInitUnicodeString(&us,L"\\Registry\\Machine\\SYSTEM\\"
258  L"CurrentControlSet\\Control");
259  InitializeObjectAttributes(&oa,&us,OBJ_CASE_INSENSITIVE,NULL,NULL);
260  status = NtOpenKey(&hKey,KEY_QUERY_VALUE,&oa);
261  if(status != STATUS_SUCCESS){
262  DebugPrintEx(status,"winx_get_windows_boot_options: cannot open %ws",us.Buffer);
263  winx_printf("winx_get_windows_boot_options: cannot open %ls: %x\n\n",us.Buffer,(UINT)status);
264  return NULL;
265  }
266 
267  /* 2. read SystemStartOptions value */
268  RtlInitUnicodeString(&us,L"SystemStartOptions");
269  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
270  NULL,0,&data_size);
271  if(status != STATUS_BUFFER_TOO_SMALL){
272  DebugPrintEx(status,"winx_get_windows_boot_options: cannot query SystemStartOptions value size");
273  winx_printf("winx_get_windows_boot_options: cannot query SystemStartOptions value size: %x\n\n",(UINT)status);
274  return NULL;
275  }
276  data_size += sizeof(short);
277  data = winx_heap_alloc(data_size);
278  if(data == NULL){
279  DebugPrint("winx_get_windows_boot_options: cannot allocate %u bytes of memory",data_size);
280  winx_printf("winx_get_windows_boot_options: cannot allocate %u bytes of memory\n\n",data_size);
281  return NULL;
282  }
283 
284  RtlZeroMemory(data,data_size);
285  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
286  data,data_size,&data_size2);
287  if(status != STATUS_SUCCESS){
288  DebugPrintEx(status,"winx_get_windows_boot_options: cannot query SystemStartOptions value");
289  winx_printf("winx_get_windows_boot_options: cannot query SystemStartOptions value: %x\n\n",(UINT)status);
290  winx_heap_free(data);
291  return NULL;
292  }
293  data_buffer = (short *)(data->Data);
294  data_length = data->DataLength >> 1;
295  if(data_length == 0) empty_value = TRUE;
296 
297  if(!empty_value){
298  data_buffer[data_length - 1] = 0;
299  buffer_size = data_length * sizeof(short);
300  } else {
301  buffer_size = 1 * sizeof(short);
302  }
303 
304  boot_options = winx_heap_alloc(buffer_size);
305  if(!boot_options){
306  DebugPrint("winx_get_windows_boot_options: cannot allocate %u bytes of memory",buffer_size);
307  winx_printf("winx_get_windows_boot_options: cannot allocate %u bytes of memory\n\n",buffer_size);
308  winx_heap_free(data);
309  return NULL;
310  }
311 
312  if(!empty_value){
313  memcpy((void *)boot_options,(void *)data_buffer,buffer_size);
314  DebugPrint("winx_get_windows_boot_options: %ls - %u",data_buffer,data_size);
315  //winx_printf("winx_get_windows_boot_options: %ls - %u\n\n",data_buffer,data_size);
316  } else {
317  boot_options[0] = 0;
318  }
319 
320  winx_heap_free(data);
321  return boot_options;
322 }

353 void MarkWindowsBootAsSuccessful(void)
354 {
355  char bootstat_file_path[MAX_PATH];
356  WINX_FILE *f_bootstat;
357  char boot_success_flag = 1;
358 
359  /*
360  * We have decided to avoid the use of related RtlXxx calls,
361  * since they're undocumented (as usually), therefore may
362  * bring us a lot of surprises.
363  */
364  if(winx_get_windows_directory(bootstat_file_path,MAX_PATH) < 0){
365  DebugPrint("MarkWindowsBootAsSuccessful(): cannot retrieve the Windows directory path");
366  winx_printf("\nMarkWindowsBootAsSuccessful(): cannot retrieve the Windows directory path\n\n");
367  winx_sleep(3000);
368  return;
369  }
370  (void)strncat(bootstat_file_path,"\\bootstat.dat",
371  MAX_PATH - strlen(bootstat_file_path) - 1);
372 
373  /* open the bootstat.dat file */
374  f_bootstat = winx_fopen(bootstat_file_path,"r+");
375  if(f_bootstat == NULL){
376  /* it seems that we have system prior to XP */
377  return;
378  }
379 
380  /* set BootSuccessFlag to 0x1 (look at BOOT_STATUS_DATA definition in ntndk.h for details) */
381  f_bootstat->woffset.QuadPart = 0xa;
382  (void)winx_fwrite(&boot_success_flag,sizeof(char),1,f_bootstat);
383 
384  /* close the file */
385  winx_fclose(f_bootstat);
386 }
