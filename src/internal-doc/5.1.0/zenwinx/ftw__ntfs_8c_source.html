1414 static winx_file_info * find_directory_by_mft_id(ULONGLONG mft_id,
1415  file_entry *f_array,unsigned long n_entries,mft_scan_parameters *sp)
1416 {

1418  ULONG lim, i, k;
1419  signed long m;
1420  BOOLEAN ascending_order;
1421 
1422  if(f_array == NULL){ /* use slow linear search */
1423  for(f = *sp->filelist; f != NULL; f = f->next){
1424  if(f->internal.BaseMftId == mft_id){

1426  return f;
1427  }
1428  if(f->next == *sp->filelist) break;
1429  }
1430  return NULL;
1431  } else { /* use fast binary search */
1432  /* Based on bsearch() algorithm copyrighted by DJ Delorie (1994). */
1433  ascending_order = (sp->mft_scan_direction == MFT_SCAN_RTL) ? TRUE : FALSE;
1434  i = 0;
1435  for(lim = n_entries; lim != 0; lim >>= 1){
1436  k = i + (lim >> 1);
1437  if(f_array[k].mft_id == mft_id){
1438  /* search for proper entry in neighbourhood of found entry */
1439  for(m = k; m >= 0; m --){
1440  if(f_array[m].mft_id != mft_id) break;
1441  }
1442  for(m = m + 1; (unsigned long)(m) < n_entries; m ++){
1443  if(f_array[m].mft_id != mft_id) break;
1444  if(wcsstr(f_array[m].f->name,L":$") == NULL)
1445  return f_array[m].f;
1446  }
1447  DebugPrint("find_directory_by_mft_id: Exit 1");
1448  return NULL;
1449  }
1450  if(ascending_order){
1451  if(mft_id > f_array[k].mft_id){
1452  i = k + 1; lim --; /* move right */
1453  } /* else move left */
1454  } else {
1455  if(mft_id < f_array[k].mft_id){
1456  i = k + 1; lim --; /* move right */
1457  } /* else move left */
1458  }
1459  }
1460  DebugPrint("find_directory_by_mft_id: Exit 2");
1461  return NULL;
1462  }
