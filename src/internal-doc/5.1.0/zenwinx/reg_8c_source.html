57 int winx_register_boot_exec_command(short *command)
58 {
59  HANDLE hKey;
60  KEY_VALUE_PARTIAL_INFORMATION *data;
61  DWORD size, value_size;
62  short *value, *pos;
63  DWORD length, i, len;
64 
65  DbgCheck1(command,"winx_register_boot_exec_command",-1);
66 
67  if(open_smss_key(&hKey) < 0) return (-1);
68  size = (wcslen(command) + 1) * sizeof(short);
69  if(read_boot_exec_value(hKey,(void **)(void *)&data,&size) < 0){
70  NtCloseSafe(hKey);
71  return (-1);
72  }
73 
74  if(data->Type != REG_MULTI_SZ){
75  DebugPrint("winx_register_boot_exec_command: BootExecute value has wrong type 0x%x",
76  data->Type);
77  winx_heap_free((void *)data);
78  NtCloseSafe(hKey);
79  return (-1);
80  }
81 
82  value = (short *)(data->Data);
83  length = (data->DataLength >> 1) - 1;
84  for(i = 0; i < length;){
85  pos = value + i;
86  //DebugPrint("%ws",pos);
87  len = wcslen(pos) + 1;
88  /* if the command is yet registered then exit */
89  if(cmd_compare(pos,command) > 0)
90  goto done;
91  i += len;
92  }
93  wcscpy(value + i,command);
94  value[i + wcslen(command) + 1] = 0;
95 
96  value_size = (i + wcslen(command) + 1 + 1) * sizeof(short);
97  if(write_boot_exec_value(hKey,(void *)(data->Data),value_size) < 0){
98  winx_heap_free((void *)data);
99  NtCloseSafe(hKey);
100  return (-1);
101  }
102 
103 done:
104  winx_heap_free((void *)data);
105  flush_smss_key(hKey);
106  NtCloseSafe(hKey);
107  return 0;
108 }

133  NtCloseSafe(hKey);

136

138  DebugPrint("winx_unregister_boot_exec_command: BootExecute value has wrong type 0x%x",
139  data->Type);

141  NtCloseSafe(hKey);
142  return (-1);
143  }
144 
145  value = (short *)(data->Data);
146  length = (data->DataLength >> 1) - 1;
147 
148  new_value_size = (length + 1) << 1;
149  new_value = winx_heap_alloc(new_value_size);
150  if(!new_value){
151  DebugPrint("winx_unregister_boot_exec_command: cannot allocate %u bytes of memory"

153  winx_heap_free((void *)data);
154  NtCloseSafe(hKey);
155  return (-1);
156  }
157 
158  memset((void *)new_value,0,new_value_size);
159  new_length = 0;
160  for(i = 0; i < length;){
161  pos = value + i;
162  //DebugPrint("%ws",pos);
163  len = wcslen(pos) + 1;
164  if(cmd_compare(pos,command) <= 0){
165  wcscpy(new_value + new_length,pos);
166  new_length += len;
167  }
168  i += len;
169  }
170  new_value[new_length] = 0;
171 
172  if(write_boot_exec_value(hKey,(void *)new_value,
173  (new_length + 1) * sizeof(short)) < 0){
174  winx_heap_free((void *)new_value);
175  winx_heap_free((void *)data);
176  NtCloseSafe(hKey);
177  return (-1);
178  }
179 
180  winx_heap_free((void *)new_value);
181  winx_heap_free((void *)data);
182  flush_smss_key(hKey);
183  NtCloseSafe(hKey);
184  return 0;
185 }

193 static int open_smss_key(HANDLE *pKey)
194 {
195  UNICODE_STRING us;
196  OBJECT_ATTRIBUTES oa;
197  NTSTATUS status;
198 
199  RtlInitUnicodeString(&us,L"\\Registry\\Machine\\SYSTEM\\"
200  L"CurrentControlSet\\Control\\Session Manager");
201  InitializeObjectAttributes(&oa,&us,OBJ_CASE_INSENSITIVE,NULL,NULL);
202  status = NtOpenKey(pKey,KEY_QUERY_VALUE | KEY_SET_VALUE,&oa);
203  if(status != STATUS_SUCCESS){
204  DebugPrintEx(status,"open_smss_key: cannot open %ws",us.Buffer);
205  return (-1);
206  }
207  return 0;
208 }

222 static int read_boot_exec_value(HANDLE hKey,void **data,DWORD *size)
223 {
224  void *data_buffer = NULL;
225  DWORD data_size = 0;
226  DWORD data_size2 = 0;
227  DWORD additional_space_size = *size;
228  UNICODE_STRING us;
229  NTSTATUS status;
230 
231  RtlInitUnicodeString(&us,L"BootExecute");
232  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
233  NULL,0,&data_size);
234  if(status != STATUS_BUFFER_TOO_SMALL){
235  DebugPrintEx(status,"read_boot_exec_value: cannot query BootExecute value size");
236  return (-1);
237  }
238  data_size += additional_space_size;
239  data_buffer = winx_heap_alloc(data_size);
240  if(data_buffer == NULL){
241  DebugPrint("read_boot_exec_value: cannot allocate %u bytes of memory",data_size);
242  return (-1);
243  }
244 
245  RtlZeroMemory(data_buffer,data_size);
246  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
247  data_buffer,data_size,&data_size2);
248  if(status != STATUS_SUCCESS){
249  DebugPrintEx(status,"read_boot_exec_value: cannot query BootExecute value");
250  winx_heap_free(data_buffer);
251  return (-1);
252  }
253 
254  *data = data_buffer;
255  *size = data_size;
256  return 0;
257 }

292 static int cmd_compare(short *reg_cmd,short *cmd)
293 {
294  short *reg_cmd_copy = NULL;
295  short *cmd_copy = NULL;
296  short *long_cmd = NULL;
297  short autocheck[] = L"autocheck ";
298  int length;
299  int result = (-1);
300 
301  /* do we have the command registered as it is? */
302  if(!wcscmp(reg_cmd,cmd))
303  return 1;
304 
305  /* allocate memory */
306  reg_cmd_copy = winx_wcsdup(reg_cmd);
307  if(reg_cmd_copy == NULL){
308  DebugPrint("cmd_compare: cannot allocate %u bytes of memory",
309  (wcslen(reg_cmd) + 1) * sizeof(short));
310  goto done;
311  }
312  cmd_copy = winx_wcsdup(cmd);
313  if(cmd_copy == NULL){
314  DebugPrint("cmd_compare: cannot allocate %u bytes of memory",
315  (wcslen(cmd) + 1) * sizeof(short));
316  goto done;
317  }
318  length = (wcslen(cmd) + wcslen(autocheck) + 1) * sizeof(short);
319  long_cmd = winx_heap_alloc(length);
320  if(long_cmd == NULL){
321  DebugPrint("cmd_compare: cannot allocate %u bytes of memory",length);
322  goto done;
323  }
324  wcscpy(long_cmd,autocheck);
325  wcscat(long_cmd,cmd);
326 
327  /* convert all strings to lowercase */
328  _wcslwr(reg_cmd_copy);
329  _wcslwr(cmd_copy);
330  _wcslwr(long_cmd);
331 
332  /* compare */
333  if(!wcscmp(reg_cmd_copy,cmd_copy) || !wcscmp(reg_cmd_copy,long_cmd)){
334  result = 1;
335  goto done;
336  }
337 
338  result = 0;
339 
340 done:
341  if(reg_cmd_copy) winx_heap_free(reg_cmd_copy);
342  if(cmd_copy) winx_heap_free(cmd_copy);
343  if(long_cmd) winx_heap_free(long_cmd);
344  return result;
345 }
