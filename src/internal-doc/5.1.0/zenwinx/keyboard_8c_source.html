75 static DWORD WINAPI kb_wait_for_input(LPVOID p)
76 {
77  LARGE_INTEGER ByteOffset;
78  IO_STATUS_BLOCK iosb;
79  KEYBOARD_INPUT_DATA kid;
80  NTSTATUS Status;
81  LARGE_INTEGER interval;
82  LARGE_INTEGER synch_interval;
83  int index;
84  char buffer[128];
85  KEYBOARD *kbd = (KEYBOARD *)p;
86 
87  /*
88  * Either debug print or winx_printf,
89  * or memory allocation aren't available here...
90  */
91 
92  kb_wait_for_input_threads ++;
93  interval.QuadPart = -((signed long)STOP_KB_WAIT_INTERVAL * 10000);
94 
95  while(!stop_kb_wait_for_input){
96  ByteOffset.QuadPart = 0;
97  /* make a read request */
98  Status = NtReadFile(kbd->hKbDevice,kbd->hKbEvent,NULL,NULL,
99  &iosb,&kid,sizeof(KEYBOARD_INPUT_DATA),&ByteOffset,0);
100  /* wait for key hits */
101  if(NT_SUCCESS(Status)){
102  do {
103  Status = NtWaitForSingleObject(kbd->hKbEvent,FALSE,&interval);
104  if(stop_kb_wait_for_input){
105  if(Status == STATUS_TIMEOUT){
106  /* cancel the pending operation */
107  Status = NtCancelIoFile(kbd->hKbDevice,&iosb);
108  if(NT_SUCCESS(Status)){
109  Status = NtWaitForSingleObject(kbd->hKbEvent,FALSE,NULL);
110  if(NT_SUCCESS(Status)) Status = iosb.Status;
111  }
112  if(!NT_SUCCESS(Status)){
113  _snprintf(buffer,sizeof(buffer),"\nNtCancelIoFile for KeyboadClass%u failed: %x!\n%s\n",
114  kbd->device_number,(UINT)Status,winx_get_error_description((ULONG)Status));
115  buffer[sizeof(buffer) - 1] = 0;
116  winx_print(buffer);
117  }
118  }
119  goto done;
120  }
121  } while(Status == STATUS_TIMEOUT);
122  if(NT_SUCCESS(Status)) Status = iosb.Status;
123  }
124  /* here we have either an input gathered or an error */
125  if(!NT_SUCCESS(Status)){
126  _snprintf(buffer,sizeof(buffer),"\nCannot read the KeyboadClass%u device: %x!\n%s\n",
127  kbd->device_number,(UINT)Status,winx_get_error_description((ULONG)Status));
128  buffer[sizeof(buffer) - 1] = 0;
129  winx_print(buffer);
130  goto done;
131  } else {
132  /* synchronize with other threads */
133  if(hKbSynchEvent){
134  synch_interval.QuadPart = MAX_WAIT_INTERVAL;
135  Status = NtWaitForSingleObject(hKbSynchEvent,FALSE,&synch_interval);
136  if(Status != WAIT_OBJECT_0){
137  _snprintf(buffer,sizeof(buffer),"\nkb_wait_for_input: synchronization failed: %x!\n%s\n",
138  (UINT)Status,winx_get_error_description((ULONG)Status));
139  buffer[sizeof(buffer) - 1] = 0;
140  winx_print(buffer);
141  }
142  }
143 
144  /* push new item to the keyboard queue */
145  if(start_index < 0 || start_index >= KB_QUEUE_LENGTH){
146  winx_print("\nkb_wait_for_input: unexpected condition #1!\n\n");
147  start_index = 0;
148  }
149  if(n_written < 0 || n_written > KB_QUEUE_LENGTH){
150  winx_print("\nkb_wait_for_input: unexpected condition #2!\n\n");
151  n_written = 0;
152  }
153 
154  index = start_index + n_written;
155  if(index >= KB_QUEUE_LENGTH)
156  index -= KB_QUEUE_LENGTH;
157 
158  if(n_written == KB_QUEUE_LENGTH)
159  start_index ++;
160  else
161  n_written ++;
162 
163  memcpy(&kids[index],&kid,sizeof(KEYBOARD_INPUT_DATA));
164 
165  /* release synchronization event */
166  if(hKbSynchEvent)
167  (void)NtSetEvent(hKbSynchEvent,NULL);
168  }
169  }
170 
171 done:
172  kb_wait_for_input_threads --;
173  winx_exit_thread(0);
174  return 0;
175 }

185 int winx_kb_init(void)
186 {
187  short event_name[64];
188  int i, j, kbdCount;
189 
190  kbdCount = query_keyboard_count();
191 
192  /* create synchronization event for safe access to kids array */
193  _snwprintf(event_name,64,L"\\winx_kb_synch_event_%u",
194  (unsigned int)(DWORD_PTR)(NtCurrentTeb()->ClientId.UniqueProcess));
195  event_name[63] = 0;
196 
197  if(hKbSynchEvent == NULL){
198  (void)winx_create_event(event_name,SynchronizationEvent,&hKbSynchEvent);
199  if(hKbSynchEvent) (void)NtSetEvent(hKbSynchEvent,NULL);
200  }
201 
202  if(hKbSynchEvent == NULL){
203  winx_printf("\nCannot create %ws event!\n\n",event_name);
204  return (-1);
205  }
206 
207  /* initialize kb array */
208  memset((void *)kb,0,sizeof(kb));
209  number_of_keyboards = 0;
210 
211  /* check all the keyboards and wait ten seconds
212  for any keyboard that fails detection.
213  required for USB devices, which can change ports */
214  for(i = 0; i < MAX_NUM_OF_KEYBOARDS; i++) {
215  if (kb_open_internal(i, kbdCount) == -1) {
216  if (i < kbdCount) {
217  winx_printf("Wait 10 seconds for keyboard initialization ");
218 
219  for(j = 0; j < 10; j++){
220  winx_sleep(1000);
221  winx_printf(".");
222  }
223  winx_printf("\n\n");
224 
225  (void)kb_open_internal(i, kbdCount);
226  }
227  }
228  }
229 
230  /* start threads waiting for user input */
231  stop_kb_wait_for_input = 0;
232  kb_wait_for_input_threads = 0;
233  for(i = 0; i < MAX_NUM_OF_KEYBOARDS; i++){
234  if(kb[i].hKbDevice == NULL) break;
235  if(winx_create_thread(kb_wait_for_input,(LPVOID)&kb[i],NULL) < 0){
236  winx_printf("\nCannot create thread gathering input from \\Device\\KeyboardClass%u\n\n",
237  kb[i].device_number);
238  /* stop all threads */
239  stop_kb_wait_for_input = 1;
240  while(kb_wait_for_input_threads)
241  winx_sleep(STOP_KB_WAIT_INTERVAL);
242  return (-1);
243  }
244  }
245 
246  if(kb[0].hKbDevice) return 0; /* success, at least one keyboard found */
247  else return (-1);
248 }

254 void kb_close(void)
255 {
256  int i;
257 
258  /*
259  * Either debug print or memory
260  * allocation calls aren't available here...
261  */
262 
263  /* stop threads waiting for user input */
264  stop_kb_wait_for_input = 1; i = 0;
265  /* 3 sec should be enough; otherwise sometimes it hangs */
266  while(kb_wait_for_input_threads && i < 30){
267  winx_sleep(STOP_KB_WAIT_INTERVAL);
268  i++;
269  }
270  if(kb_wait_for_input_threads){
271  winx_printf("Keyboards polling terminated forcibly...\n");
272  winx_sleep(2000);
273  }
274 
275  for(i = 0; i < MAX_NUM_OF_KEYBOARDS; i++){
276  if(kb[i].hKbDevice == NULL) break;
277  NtCloseSafe(kb[i].hKbDevice);
278  NtCloseSafe(kb[i].hKbEvent);
279  /* don't reset device_number member here */
280  number_of_keyboards --;
281  }
282 
283  /* destroy synchronization event */
284  winx_destroy_event(hKbSynchEvent);
285 }

296 int kb_read(PKEYBOARD_INPUT_DATA pKID,int msec_timeout)
297 {
298  int attempts = 0;
299  ULONGLONG xtime = 0;
300  LARGE_INTEGER synch_interval;
301  NTSTATUS Status;
302 
303  DbgCheck1(pKID,"kb_read",-1);
304 
305  if(msec_timeout != INFINITE){
306  attempts = msec_timeout / MAX_TYPING_DELAY + 1;
307  xtime = winx_xtime();
308  }
309 
310  while(number_of_keyboards){
311  /* synchronize with other threads */
312  if(hKbSynchEvent){
313  synch_interval.QuadPart = MAX_WAIT_INTERVAL;
314  Status = NtWaitForSingleObject(hKbSynchEvent,FALSE,&synch_interval);
315  if(Status != WAIT_OBJECT_0){
316  winx_printf("\nkb_read: synchronization failed: 0x%x\n",(UINT)Status);
317  winx_printf("%s\n\n",winx_get_error_description((ULONG)Status));
318  }
319  }
320 
321  /* pop item from the keyboard queue */
322  if(n_written > 0){
323  memcpy(pKID,&kids[start_index],sizeof(KEYBOARD_INPUT_DATA));
324  start_index ++;
325  if(start_index >= KB_QUEUE_LENGTH)
326  start_index = 0;
327  n_written --;
328  if(hKbSynchEvent)
329  (void)NtSetEvent(hKbSynchEvent,NULL);
330  return 0;
331  }
332 
333  /* release synchronization event */
334  if(hKbSynchEvent)
335  (void)NtSetEvent(hKbSynchEvent,NULL);
336 
337  winx_sleep(MAX_TYPING_DELAY);
338  if(msec_timeout != INFINITE){
339  attempts --;
340  if(attempts == 0) break;
341  if(xtime && (winx_xtime() - xtime >= msec_timeout)) break;
342  }
343  }
344  return (-1);
345 }

360 static int kb_open_internal(int device_number, int kbd_count)
361 {
362  short device_name[32];
363  short event_name[32];
364  UNICODE_STRING uStr;
365  OBJECT_ATTRIBUTES ObjectAttributes;
366  IO_STATUS_BLOCK IoStatusBlock;
367  NTSTATUS Status;
368  HANDLE hKbDevice = NULL;
369  HANDLE hKbEvent = NULL;
370  int i;
371 
372  (void)_snwprintf(device_name,32,L"\\Device\\KeyboardClass%u",device_number);
373  device_name[31] = 0;
374  RtlInitUnicodeString(&uStr,device_name);
375  InitializeObjectAttributes(&ObjectAttributes,&uStr,OBJ_CASE_INSENSITIVE,NULL,NULL);
376  Status = NtCreateFile(&hKbDevice,
377  GENERIC_READ | FILE_RESERVE_OPFILTER | FILE_READ_ATTRIBUTES/*0x80100080*/,
378  &ObjectAttributes,&IoStatusBlock,NULL,FILE_ATTRIBUTE_NORMAL/*0x80*/,
379  0,FILE_OPEN/*1*/,FILE_DIRECTORY_FILE/*1*/,NULL,0);
380  if(!NT_SUCCESS(Status)){
381  if(device_number < kbd_count){
382  DebugPrintEx(Status,"Cannot open the keyboard %ws",device_name);
383  winx_printf("\nCannot open the keyboard %ws: %x!\n",
384  device_name,(UINT)Status);
385  winx_printf("%s\n",winx_get_error_description((ULONG)Status));
386  }
387  return (-1);
388  }
389 
390  /* ensure that we have opened a really connected keyboard */
391  if(kb_check(hKbDevice) < 0){
392  DebugPrintEx(Status,"Invalid keyboard device %ws",device_name);
393  winx_printf("\nInvalid keyboard device %ws: %x!\n",device_name,(UINT)Status);
394  winx_printf("%s\n",winx_get_error_description((ULONG)Status));
395  NtCloseSafe(hKbDevice);
396  return (-1);
397  }
398 
399  /* create a special event object for internal use */
400  (void)_snwprintf(event_name,32,L"\\kb_event%u",device_number);
401  event_name[31] = 0;
402  RtlInitUnicodeString(&uStr,event_name);
403  InitializeObjectAttributes(&ObjectAttributes,&uStr,0,NULL,NULL);
404  Status = NtCreateEvent(&hKbEvent,STANDARD_RIGHTS_ALL | 0x1ff/*0x1f01ff*/,
405  &ObjectAttributes,SynchronizationEvent,0/*FALSE*/);
406  if(!NT_SUCCESS(Status)){
407  NtCloseSafe(hKbDevice);
408  DebugPrintEx(Status,"Cannot create kb_event%u",device_number);
409  winx_printf("\nCannot create kb_event%u: %x!\n",device_number,(UINT)Status);
410  winx_printf("%s\n",winx_get_error_description((ULONG)Status));
411  return (-1);
412  }
413 
414  /* add information to kb array */
415  for(i = 0; i < MAX_NUM_OF_KEYBOARDS; i++){
416  if(kb[i].hKbDevice == NULL){
417  kb[i].hKbDevice = hKbDevice;
418  kb[i].hKbEvent = hKbEvent;
419  kb[i].device_number = device_number;
420  number_of_keyboards ++;
421  winx_printf("Keyboard device found: %ws.\n",device_name);
422  return 0;
423  }
424  }
425 
426  winx_printf("\nkb array is full!\n");
427  return (-1);
428 }

467 static int kb_check(HANDLE hKbDevice)
468 {
469  USHORT LedFlags;
470  NTSTATUS Status;
471  IO_STATUS_BLOCK iosb;
472  KEYBOARD_INDICATOR_PARAMETERS kip;
473  int i;
474 
475  /* try to get LED flags */
476  RtlZeroMemory(&kip,sizeof(KEYBOARD_INDICATOR_PARAMETERS));
477  Status = NtDeviceIoControlFile(hKbDevice,NULL,NULL,NULL,
478  &iosb,IOCTL_KEYBOARD_QUERY_INDICATORS,NULL,0,
479  &kip,sizeof(KEYBOARD_INDICATOR_PARAMETERS));
480  if(NT_SUCCESS(Status)){
481  Status = NtWaitForSingleObject(hKbDevice,FALSE,NULL);
482  if(NT_SUCCESS(Status)) Status = iosb.Status;
483  }
484  if(!NT_SUCCESS(Status) || Status == STATUS_PENDING) return (-1);
485 
486  LedFlags = kip.LedFlags;
487 
488  /* light up LED's */
489  for(i = 0; i < LIGHTING_REPEAT_COUNT; i++){
490  (void)kb_light_up_indicators(hKbDevice,KEYBOARD_NUM_LOCK_ON);
491  winx_sleep(100);
492  (void)kb_light_up_indicators(hKbDevice,KEYBOARD_CAPS_LOCK_ON);
493  winx_sleep(100);
494  (void)kb_light_up_indicators(hKbDevice,KEYBOARD_SCROLL_LOCK_ON);
495  winx_sleep(100);
496  }
497 
498  (void)kb_light_up_indicators(hKbDevice,LedFlags);
499  return 0;

507 static int query_keyboard_count(void)
508 {
509  UNICODE_STRING us;
510  OBJECT_ATTRIBUTES oa;
511  NTSTATUS status;
512  HANDLE hKey;
513  KEY_VALUE_PARTIAL_INFORMATION *data_buffer = NULL;
514  DWORD data_size = 0;
515  DWORD data_size2 = 0;
516  int kbdCount = 2;
517 
518  RtlInitUnicodeString(&us,L"\\Registry\\Machine\\SYSTEM\\"
519  L"CurrentControlSet\\Services\\Kbdclass\\Enum");
520  InitializeObjectAttributes(&oa,&us,OBJ_CASE_INSENSITIVE,NULL,NULL);
521  status = NtOpenKey(&hKey,KEY_QUERY_VALUE,&oa);
522  if(status != STATUS_SUCCESS){
523  DebugPrintEx(status,"query_keyboard_count: cannot open %ws",us.Buffer);
524  return kbdCount;
525  }
526 
527  RtlInitUnicodeString(&us,L"Count");
528  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
529  NULL,0,&data_size);
530  if(status != STATUS_BUFFER_TOO_SMALL){
531  DebugPrintEx(status,"query_keyboard_count: cannot query Count value size");
532  NtCloseSafe(hKey);
533  return kbdCount;
534  }
535  data_buffer = (KEY_VALUE_PARTIAL_INFORMATION *)winx_heap_alloc(data_size);
536  if(data_buffer == NULL){
537  DebugPrint("query_keyboard_count: cannot allocate %u bytes of memory",data_size);
538  NtCloseSafe(hKey);
539  return kbdCount;
540  }
541 
542  RtlZeroMemory(data_buffer,data_size);
543  status = NtQueryValueKey(hKey,&us,KeyValuePartialInformation,
544  data_buffer,data_size,&data_size2);
545  if(status != STATUS_SUCCESS){
546  DebugPrintEx(status,"query_keyboard_count: cannot query Count value");
547  winx_heap_free(data_buffer);
548  NtCloseSafe(hKey);
549  return kbdCount;
550  }
551 
552  if(data_buffer->Type != REG_DWORD){
553  DebugPrint("query_keyboard_count: Count value has wrong type 0x%x",
554  data_buffer->Type);
555  winx_heap_free(data_buffer);
556  NtCloseSafe(hKey);
557  return kbdCount;
558  }
559 
560  kbdCount = (int)*(DWORD *)data_buffer->Data;
561  DebugPrint("query_keyboard_count: keyboard count is %u",kbdCount);
562 
563  winx_heap_free(data_buffer);
564  NtCloseSafe(hKey);
565 
566  return kbdCount;
567 }
