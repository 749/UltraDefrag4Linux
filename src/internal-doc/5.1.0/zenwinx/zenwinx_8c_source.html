108 int winx_init_library(void *peb)
109 {
110  PRTL_USER_PROCESS_PARAMETERS pp;
111 
112  /* normalize and get the process parameters */
113  if(peb){
114  pp = RtlNormalizeProcessParams(((PPEB)peb)->ProcessParameters);
115  /* breakpoint if we were requested to do so */
116  if(pp){
117  if(pp->DebugFlags)
118  DbgBreakPoint();
119  }
120  }
121 
122  winx_init_case_tables();
123  if(winx_create_global_heap() < 0)
124  return (-1);
125  if(winx_dbg_init() < 0)
126  return (-1);
127  initialization_failed = 0;
128  return 0;
129 }

182 void winx_exit(int exit_code)
183 {
184  NTSTATUS Status;
185 
186  kb_close();
187  winx_flush_dbg_log();
188  winx_unload_library();
189  Status = NtTerminateProcess(NtCurrentProcess(),exit_code);
190  if(!NT_SUCCESS(Status)){
191  print_post_scriptum("winx_exit: cannot terminate process",Status);
192  }
193 }
