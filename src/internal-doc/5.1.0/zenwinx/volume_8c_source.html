68 int winx_get_drive_type(char letter)
69 {
70  wchar_t link_name[] = L"\\??\\A:";
71  #define MAX_TARGET_LENGTH 256
72  short link_target[MAX_TARGET_LENGTH];
73  PROCESS_DEVICEMAP_INFORMATION *ppdi;
74  FILE_FS_DEVICE_INFORMATION *pffdi;
75  IO_STATUS_BLOCK iosb;
76  NTSTATUS Status;
77  int drive_type;
78  HANDLE hRoot;
79 
80  /* An additional checks for DFS were suggested by Stefan Pendl (pendl2megabit@yahoo.de). */
81  /* DFS shares have DRIVE_NO_ROOT_DIR type though they are actually remote. */
82 
83  letter = winx_toupper(letter); /* possibly required for w2k */
84  if(letter < 'A' || letter > 'Z'){
85  DebugPrint("winx_get_drive_type: invalid letter %c",letter);
86  return (-1);
87  }
88 
89  /* check for the drive existence */
90  link_name[4] = (short)letter;
91  if(winx_query_symbolic_link(link_name,link_target,MAX_TARGET_LENGTH) < 0)
92  return (-1);
93 
94  /* check for an assignment made by subst command */
95  if(wcsstr(link_target,L"\\??\\") == (wchar_t *)link_target)
96  return DRIVE_ASSIGNED_BY_SUBST_COMMAND;
97 
98  /* check for classical floppies */
99  if(wcsstr(link_target,L"Floppy"))
100  return DRIVE_REMOVABLE;
101 
102  /* allocate memory */
103  ppdi = winx_heap_alloc(sizeof(PROCESS_DEVICEMAP_INFORMATION));
104  if(!ppdi){
105  DebugPrint("winx_get_drive_type: cannot allocate %u bytes of memory",
106  sizeof(PROCESS_DEVICEMAP_INFORMATION));
107  return (-1);
108  }
109  pffdi = winx_heap_alloc(sizeof(FILE_FS_DEVICE_INFORMATION));
110  if(!pffdi){
111  winx_heap_free(ppdi);
112  DebugPrint("winx_get_drive_type: cannot allocate %u bytes of memory",
113  sizeof(FILE_FS_DEVICE_INFORMATION));
114  return (-1);
115  }
116 
117  /* try to define exactly which type has a specified drive (w2k+) */
118  RtlZeroMemory(ppdi,sizeof(PROCESS_DEVICEMAP_INFORMATION));
119  Status = NtQueryInformationProcess(NtCurrentProcess(),
120  ProcessDeviceMap,ppdi,
121  sizeof(PROCESS_DEVICEMAP_INFORMATION),
122  NULL);
123  if(NT_SUCCESS(Status)){
124  drive_type = (int)ppdi->Query.DriveType[letter - 'A'];
125  /*
126  * Type DRIVE_NO_ROOT_DIR have the following drives:
127  * 1. assigned by subst command
128  * 2. SCSI external drives
129  * 3. RAID volumes
130  * 4. DFS shares
131  * We need additional checks to know exactly.
132  */
133  if(drive_type != DRIVE_NO_ROOT_DIR){
134  winx_heap_free(ppdi);
135  winx_heap_free(pffdi);
136  return drive_type;
137  }
138  } else {
139  if(Status != STATUS_INVALID_INFO_CLASS){ /* exclude common NT4 error code */
140  /* exclude common NT4 Terminal Server Edition error code */
141  if(Status != STATUS_INFO_LENGTH_MISMATCH){
142  DebugPrintEx(Status,"winx_get_drive_type: cannot get device map");
143  winx_heap_free(ppdi);
144  winx_heap_free(pffdi);
145  return (-1);
146  }
147  }
148  }
149 
150  /* try to define exactly again which type has a specified drive (nt4+) */
151  /* note that the drive motor can be powered on during this check */
152  hRoot = OpenRootDirectory(letter);
153  if(hRoot == NULL){
154  winx_heap_free(ppdi);
155  winx_heap_free(pffdi);
156  return (-1);
157  }
158  RtlZeroMemory(pffdi,sizeof(FILE_FS_DEVICE_INFORMATION));
159  Status = NtQueryVolumeInformationFile(hRoot,&iosb,
160  pffdi,sizeof(FILE_FS_DEVICE_INFORMATION),
161  FileFsDeviceInformation);
162  NtClose(hRoot);
163  if(!NT_SUCCESS(Status)){
164  DebugPrintEx(Status,"winx_get_drive_type: cannot get volume type for \'%c\'",letter);
165  winx_heap_free(ppdi);
166  winx_heap_free(pffdi);
167  return (-1);
168  }
169 
170  /* detect remote/cd/dvd/unknown drives */
171  if(pffdi->Characteristics & FILE_REMOTE_DEVICE){
172  winx_heap_free(ppdi);
173  winx_heap_free(pffdi);
174  return DRIVE_REMOTE;
175  }
176  switch(pffdi->DeviceType){
177  case FILE_DEVICE_CD_ROM:
178  case FILE_DEVICE_CD_ROM_FILE_SYSTEM:
179  case FILE_DEVICE_DVD: /* ? */
180  winx_heap_free(ppdi);
181  winx_heap_free(pffdi);
182  return DRIVE_CDROM;
183  case FILE_DEVICE_NETWORK_FILE_SYSTEM:
184  case FILE_DEVICE_NETWORK: /* ? */
185  case FILE_DEVICE_NETWORK_BROWSER: /* ? */
186  case FILE_DEVICE_DFS_FILE_SYSTEM:
187  case FILE_DEVICE_DFS_VOLUME:
188  case FILE_DEVICE_DFS:
189  winx_heap_free(ppdi);
190  winx_heap_free(pffdi);
191  return DRIVE_REMOTE;
192  case FILE_DEVICE_UNKNOWN:
193  winx_heap_free(ppdi);
194  winx_heap_free(pffdi);
195  return DRIVE_UNKNOWN;
196  }
197 
198  /* detect removable disks */
199  if(pffdi->Characteristics & FILE_REMOVABLE_MEDIA){
200  winx_heap_free(ppdi);
201  winx_heap_free(pffdi);
202  return DRIVE_REMOVABLE;
203  }
204 
205  /* detect fixed disks */
206  switch(pffdi->DeviceType){
207  case FILE_DEVICE_DISK:
208  case FILE_DEVICE_FILE_SYSTEM: /* ? */
209  /*case FILE_DEVICE_VIRTUAL_DISK:*/
210  /*case FILE_DEVICE_MASS_STORAGE:*/
211  case FILE_DEVICE_DISK_FILE_SYSTEM:
212  winx_heap_free(ppdi);
213  winx_heap_free(pffdi);
214  return DRIVE_FIXED;
215  }
216 
217  /* nothing detected => drive type is unknown */
218  winx_heap_free(ppdi);
219  winx_heap_free(pffdi);
220  return DRIVE_UNKNOWN;
221 }

283 static int IsFatPartition(BPB *bpb,winx_volume_information *v)
284 {
285  char signature[9];
286  BOOL fat_found = FALSE;
287  ULONG RootDirSectors; /* USHORT ? */
288  ULONG FatSectors;
289  ULONG TotalSectors;
290  ULONG DataSectors;
291  ULONG CountOfClusters;
292 
293  /* search for FAT signatures */
294  signature[8] = 0;
295  memcpy((void *)signature,(void *)bpb->Fat1x.BS_FilSysType,8);
296  if(strstr(signature,"FAT")) fat_found = TRUE;
297  else {
298  memcpy((void *)signature,(void *)bpb->Fat32.BS_FilSysType,8);
299  if(strstr(signature,"FAT")) fat_found = TRUE;
300  }
301  if(!fat_found)
302  return (-1);
303 
304  /* determine which type of FAT we have */
305  RootDirSectors = ((bpb->RootDirEnts * 32) + (bpb->BytesPerSec - 1)) / bpb->BytesPerSec;
306 
307  if(bpb->FAT16sectors) FatSectors = (ULONG)(bpb->FAT16sectors);
308  else FatSectors = bpb->Fat32.FAT32sectors;
309 
310  if(bpb->FAT16totalsectors) TotalSectors = bpb->FAT16totalsectors;
311  else TotalSectors = bpb->FAT32totalsectors;
312 
313  DataSectors = TotalSectors - (bpb->ReservedSectors + (bpb->NumFATs * FatSectors) + RootDirSectors);
314  CountOfClusters = DataSectors / bpb->SecPerCluster;
315 
316  if(CountOfClusters < 4085) {
317  /* Volume is FAT12 */
318  strcpy(v->fs_name,"FAT12");
319  return 0;
320  } else if(CountOfClusters < 65525) {
321  /* Volume is FAT16 */
322  strcpy(v->fs_name,"FAT16");
323  return 0;
324  } else {
325  /* Volume is FAT32 */
326  strcpy(v->fs_name,"FAT32");
327  }
328 
329  /* save FAT32 version */
330  v->fat32_mj_version = (ULONG)((bpb->Fat32.BPB_FSVer >> 8) & 0xFF);
331  v->fat32_mn_version = (ULONG)(bpb->Fat32.BPB_FSVer & 0xFF);
332  return 0;
333 }

429 static int get_filesystem_name(HANDLE hRoot,winx_volume_information *v)
430 {
431  FILE_FS_ATTRIBUTE_INFORMATION *pfa;
432  int fs_attr_info_size;
433  IO_STATUS_BLOCK IoStatusBlock;
434  NTSTATUS Status;
435  short fs_name[MAX_FS_NAME_LENGTH + 1];
436  int length;
437 
438 #ifdef GET_FS_NAME_FROM_THE_FIRST_SECTOR
439  void *first_sector;
440  BPB *bpb;
441 
442  /*
443  * We're using a low level analysis
444  * of the first sector firstly, because
445  * this method is able to detect exactly
446  * the type of the file system. Also,
447  * it well distinguishes different
448  * editions of FAT.
449  */
450  if(v->bytes_per_sector >= sizeof(BPB)){
451  /* allocate memory */
452  first_sector = winx_heap_alloc(v->bytes_per_sector);
453  if(first_sector == NULL){
454  DebugPrint("winx_get_volume_information: cannot allocate %u bytes of memory",
455  v->bytes_per_sector);
456  } else {
457  if(read_first_sector(first_sector,v) < 0){
458  winx_heap_free(first_sector);
459  } else {
460  bpb = (BPB *)first_sector;
461  if(IsNtfsPartition(bpb,v) >= 0){
462  winx_heap_free(first_sector);
463  return 0;
464  }
465  if(IsFatPartition(bpb,v) >= 0){
466  winx_heap_free(first_sector);
467  return 0;
468  }
469  winx_heap_free(first_sector);
470  }
471  }
472  }
473 #endif /* GET_FS_NAME_FROM_THE_FIRST_SECTOR */
474 
475  /*
476  * If direct sector analysis failed,
477  * then get file system name through
478  * FILE_FS_ATTRIBUTE_INFORMATION.
479  */
480  fs_attr_info_size = MAX_PATH * sizeof(WCHAR) + sizeof(FILE_FS_ATTRIBUTE_INFORMATION);
481  pfa = winx_heap_alloc(fs_attr_info_size);
482  if(pfa == NULL){
483  DebugPrint("winx_get_volume_information: cannot allocate %u bytes of memory",
484  fs_attr_info_size);
485  return(-1);
486  }
487 
488  RtlZeroMemory(pfa,fs_attr_info_size);
489  Status = NtQueryVolumeInformationFile(hRoot,&IoStatusBlock,pfa,
490  fs_attr_info_size,FileFsAttributeInformation);
491  if(!NT_SUCCESS(Status)){
492  DebugPrintEx(Status,"winx_get_volume_information: cannot get file system name of drive %c:",
493  v->volume_letter);
494  winx_heap_free(pfa);
495  return (-1);
496  }
497 
498  /*
499  * pfa->FileSystemName.Buffer may be not NULL terminated
500  * (theoretically), so name extraction is more tricky
501  * than it should be.
502  */
503  length = min(MAX_FS_NAME_LENGTH,pfa->FileSystemNameLength / sizeof(short));
504  wcsncpy(fs_name,pfa->FileSystemName,length);
505  fs_name[length] = 0;
506  _snprintf(v->fs_name,MAX_FS_NAME_LENGTH,"%ws",fs_name);
507  v->fs_name[MAX_FS_NAME_LENGTH] = 0;
508 
509  /* cleanup */
510  winx_heap_free(pfa);
511  return 0;
512 }

622 int winx_get_volume_information(char volume_letter,winx_volume_information *v)
623 {
624  HANDLE hRoot;
625 
626  /* check input data correctness */
627  if(v == NULL)
628  return (-1);
629 
630  /* ensure that it will work on w2k */
631  volume_letter = winx_toupper(volume_letter);
632 
633  /* reset all fields of the structure, except of volume_letter */
634  memset(v,0,sizeof(winx_volume_information));
635  v->volume_letter = volume_letter;
636 
637  if(volume_letter < 'A' || volume_letter > 'Z')
638  return (-1);
639 
640  /* open root directory */
641  hRoot = OpenRootDirectory(volume_letter);
642  if(hRoot == NULL)
643  return (-1);
644 
645  /* get drive geometry */
646  if(get_drive_geometry(hRoot,v) < 0){
647  NtClose(hRoot);
648  return (-1);
649  }
650 
651  /* get the name of contained file system */
652  if(get_filesystem_name(hRoot,v) < 0){
653  NtClose(hRoot);
654  return (-1);
655  }
656 
657  /* get name of the volume */
658  get_volume_label(hRoot,v);
659 
660  /* get NTFS data */

665  volume_letter);
666  }
667  }
668

670  get_volume_dirty_flag(v);
671 
672  NtClose(hRoot);
673  return 0;

721  #define BITMAPSIZE (BITMAPBYTES + 2 * sizeof(ULONGLONG))

764  return rlist;
765  } else {
766  winx_list_destroy((list_entry **)(void *)&rlist);
767  return NULL;
768  }
769  }
770 
771  /* scan through the returned bitmap info */
772  start = bitmap->StartLcn;
773  for(i = 0; i < min(bitmap->ClustersToEndOfVol, 8 * BITMAPBYTES); i++){
774  if(!(bitmap->Map[ i/8 ] & bitshift[ i % 8 ])){
775  /* cluster is free */
776  if(free_rgn_start == LLINVALID)
777  free_rgn_start = start + i;
778  } else {
779  /* cluster isn't free */
780  if(free_rgn_start != LLINVALID){
781  /* add free region to the list */
782  rgn = (winx_volume_region *)winx_list_insert_item((list_entry **)(void *)&rlist,
783  (list_entry *)rgn,sizeof(winx_volume_region));
784  if(rgn == NULL){
785  DebugPrint("winx_get_free_volume_regions: cannot allocate %u bytes of memory",
786  sizeof(winx_volume_region));
787  /* return if partial results aren't allowed */
788  if(!(flags & WINX_GVR_ALLOW_PARTIAL_SCAN))
789  goto fail;
790  } else {
791  rgn->lcn = free_rgn_start;
792  rgn->length = start + i - free_rgn_start;
793  if(cb != NULL){
794  if(cb(rgn,user_defined_data))
795  goto done;
796  }
797  }
798  free_rgn_start = LLINVALID;
799  }
800  }
801  }
802 
803  /* go to the next portion of data */
804  next = bitmap->StartLcn + i;
805  } while(status != STATUS_SUCCESS);
806 
807  if(free_rgn_start != LLINVALID){
808  /* add free region to the list */
809  rgn = (winx_volume_region *)winx_list_insert_item((list_entry **)(void *)&rlist,
810  (list_entry *)rgn,sizeof(winx_volume_region));
811  if(rgn == NULL){

813  sizeof(winx_volume_region));

822  goto done;
823  }
824  }

826  }

830  winx_fclose(f);
831  winx_heap_free(bitmap);

833

850 winx_volume_region *winx_add_volume_region(winx_volume_region *rlist,
851  ULONGLONG lcn,ULONGLONG length)
852 {
853  winx_volume_region *r, *rnext, *rprev = NULL;
854 
855  /* don't insert regions of zero length */
856  if(length == 0) return rlist;
857 
858  for(r = rlist; r; r = r->next){
859  if(r->lcn > lcn){
860  if(r != rlist) rprev = r->prev;
861  break;
862  }
863  if(r->next == rlist){
864  rprev = r;
865  break;
866  }
867  }
868 
869  /* hits the new region previous one? */
870  if(rprev){
871  if(rprev->lcn + rprev->length == lcn){
872  rprev->length += length;
873  if(rprev->lcn + rprev->length == rprev->next->lcn){
874  rprev->length += rprev->next->length;
875  winx_list_remove_item((list_entry **)(void *)&rlist,
876  (list_entry *)rprev->next);
877  }
878  return rlist;
879  }
880  }
881 
882  /* hits the new region the next one? */
883  if(rlist){
884  if(rprev == NULL) rnext = rlist;
885  else rnext = rprev->next;
886  if(lcn + length == rnext->lcn){
887  rnext->lcn = lcn;
888  rnext->length += length;
889  return rlist;
890  }
891  }
892 
893  r = (winx_volume_region *)winx_list_insert_item((list_entry **)(void *)&rlist,
894  (list_entry *)rprev,sizeof(winx_volume_region));
895  if(r == NULL)
896  return rlist;
897 
898  r->lcn = lcn;
899  r->length = length;
900  return rlist;
901 }

910 winx_volume_region *winx_sub_volume_region(winx_volume_region *rlist,
911  ULONGLONG lcn,ULONGLONG length)
912 {
913  winx_volume_region *r, *head, *next = NULL;
914  ULONGLONG remaining_clusters;
915  ULONGLONG new_lcn, new_length;
916 
917  remaining_clusters = length;
918  for(r = rlist; r && remaining_clusters; r = next){
919  head = rlist;
920  next = r->next;
921  if(r->lcn >= lcn + length) break;
922  if(r->lcn + r->length > lcn){
923  /* sure, at least a part of region is inside a specified range */
924  if(r->lcn >= lcn && (r->lcn + r->length) <= (lcn + length)){
925  /*
926  * list entry is inside a specified range
927  * |--------------------|
928  * |-r-|
929  */
930  remaining_clusters -= r->length;
931  winx_list_remove_item((list_entry **)(void *)&rlist,
932  (list_entry *)r);
933  goto next_region;
934  }
935  if(r->lcn < lcn && (r->lcn + r->length) > lcn && \
936  (r->lcn + r->length) <= (lcn + length)){
937  /*
938  * cut the right side of the list entry
939  * |--------------------|
940  * |----r----|
941  */
942  r->length = lcn - r->lcn;
943  goto next_region;
944  }
945  if(r->lcn >= lcn && r->lcn < (lcn + length)){
946  /*
947  * cut the left side of the list entry
948  * |--------------------|
949  * |----r----|
950  */
951  new_lcn = lcn + length;
952  new_length = r->lcn + r->length - (lcn + length);
953  winx_list_remove_item((list_entry **)(void *)&rlist,
954  (list_entry *)r);
955  rlist = winx_add_volume_region(rlist,new_lcn,new_length);
956  goto next_region;
957  }
958  if(r->lcn < lcn && (r->lcn + r->length) > (lcn + length)){
959  /*
960  * specified range is inside list entry
961  * |----|
962  * |-------r--------|
963  */
964  new_lcn = lcn + length;
965  new_length = r->lcn + r->length - (lcn + length);
966  r->length = lcn - r->lcn;
967  rlist = winx_add_volume_region(rlist,new_lcn,new_length);
968  goto next_region;
969  }
970  }
971 next_region:
972  if(rlist == NULL) break;
973  if(next == head) break;
974  }
975  return rlist;
976 }
