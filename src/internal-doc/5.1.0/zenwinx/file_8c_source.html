205 size_t winx_fwrite(const void *buffer,size_t size,size_t count,WINX_FILE *f)
206 {
207  LARGE_INTEGER nwd_offset; /* offset of data not written yet, in file */
208  LARGE_INTEGER new_offset; /* current f->woffset */
209  size_t bytes, result;
210 
211  if(buffer == NULL || f == NULL)
212  return 0;
213 
214  /*
215  * Check whether the file was
216  * opened for buffered access or not.
217  */
218  bytes = size * count;
219  if(f->io_buffer == NULL || f->io_buffer_size == 0){
220  f->io_buffer_offset = 0;
221  f->wboffset.QuadPart += bytes;
222  return winx_fwrite_helper(buffer,size,count,f);
223  }
224 
225  /* check whether file pointer has been adjusted or not */
226  nwd_offset.QuadPart = f->wboffset.QuadPart - f->io_buffer_offset;
227  new_offset.QuadPart = f->woffset.QuadPart;
228  if(new_offset.QuadPart != nwd_offset.QuadPart){
229  /* flush buffer */
230  f->woffset.QuadPart = nwd_offset.QuadPart;
231  result = winx_fwrite_helper(f->io_buffer,1,f->io_buffer_offset,f);
232  f->io_buffer_offset = 0;
233  /* update file pointer */
234  f->wboffset.QuadPart = f->woffset.QuadPart = new_offset.QuadPart;
235  if(result == 0){
236  /* write request failed */
237  return 0;
238  }
239  }
240 
241  /* check whether the buffer is full or not */
242  if(bytes > f->io_buffer_size - f->io_buffer_offset && f->io_buffer_offset){
243  /* flush buffer */
244  result = winx_fwrite_helper(f->io_buffer,1,f->io_buffer_offset,f);
245  f->io_buffer_offset = 0;
246  if(result == 0){
247  /* write request failed */
248  return 0;
249  }
250  }
251 
252  /* check whether the buffer has sufficient size or not */
253  if(bytes >= f->io_buffer_size){
254  f->wboffset.QuadPart += bytes;
255  return winx_fwrite_helper(buffer,size,count,f);
256  }
257 
258  /* append new data to the buffer */
259  memcpy((char *)f->io_buffer + f->io_buffer_offset,buffer,bytes);
260  f->io_buffer_offset += bytes;
261  f->wboffset.QuadPart += bytes;
262  return count;
263 }

505  return NULL;

509  if(n_read == 0 || n_read > length){
510  winx_heap_free(contents);
511  winx_fclose(f);
512  return NULL;
513  }
514 
515  if(bytes_read) *bytes_read = n_read;
516  winx_fclose(f);
517  return contents;

568 NTSTATUS winx_defrag_fopen(winx_file_info *f,int action,HANDLE *phandle)
569 {
570  UNICODE_STRING us;
571  OBJECT_ATTRIBUTES oa;
572  IO_STATUS_BLOCK iosb;
573  NTSTATUS status;
574  int win_version = winx_get_os_version();
575  ACCESS_MASK access_rights = SYNCHRONIZE;
576  ULONG flags = FILE_SYNCHRONOUS_IO_NONALERT;
577  int length, i;
578  char volume_letter;
579  wchar_t *path;
580  wchar_t buffer[MAX_PATH + 1];
581 
582  if(f == NULL || phandle == NULL)
583  return STATUS_INVALID_PARAMETER;
584 
585  if(f->path == NULL)
586  return STATUS_INVALID_PARAMETER;
587 
588  if(f->path[0] == 0)
589  return STATUS_INVALID_PARAMETER;
590 
591  if(is_directory(f)){
592  flags |= FILE_OPEN_FOR_BACKUP_INTENT;
593  } else {
594  flags |= FILE_NO_INTERMEDIATE_BUFFERING;
595 
596  if(win_version >= WINDOWS_VISTA)
597  flags |= FILE_NON_DIRECTORY_FILE;
598  }
599 
600  if(is_reparse_point(f)){
601  /* open the point itself, not its target */
602  flags |= FILE_OPEN_REPARSE_POINT;
603  }
604 
605  /*
606  * All files except of internal NTFS files
607  * can be successfully opened with FILE_GENERIC_READ | SYNCHRONIZE
608  * access rights on all of the supported versions of Windows.
609  * To open internal NTFS files including $mft, we use more restricted rights.
610  */
611  if(win_version <= WINDOWS_2K){
612  /* on Windows NT and Windows 2000 */
613  if(is_encrypted(f)){
614  /* encrypted files may require read access */
615  access_rights |= FILE_GENERIC_READ;
616  } else {
617  /*
618  * All other files can be successfully opened with a single
619  * SYNCHRONIZE access. More advanced FILE_GENERIC_READ
620  * rights may prevent opening of internal NTFS files on w2k.
621  */
622  }
623  } else if(win_version == WINDOWS_XP || win_version == WINDOWS_2K3){
624  /* On Windows XP and Windows Server 2003 */
625  /*
626  * All files can be opened with a single SYNCHRONIZE access.
627  * More advanced FILE_GENERIC_READ rights prevent opening
628  * of $mft file as well as other internal NTFS files.
629  * http://forum.sysinternals.com/topic23950.html
630  */
631  } else if(win_version >= WINDOWS_VISTA){
632  /* On Windows Vista and Windows 7 */
633  /*
634  * $Mft may require more advanced rights,
635  * than a single SYNCHRONIZE.
636  */
637  access_rights |= FILE_READ_ATTRIBUTES;
638  }
639 
640  /* root folder needs FILE_READ_ATTRIBUTES to successfully retrieve FileBasicInformation,
641  see http://msdn.microsoft.com/en-us/library/ff567052(VS.85).aspx */
642  if(action == WINX_OPEN_FOR_BASIC_INFO)
643  access_rights |= FILE_READ_ATTRIBUTES;
644 
645  /*
646  * FILE_READ_ATTRIBUTES may also be needed
647  * for bitmaps on Windows XP as stated in:
648  * http://www.microsoft.com/whdc/archive/2kuptoXP.mspx
649  * However, nonresident bitmaps seem to be extraordinary.
650  */
651 
652  /*
653  * Handle special cases, according to
654  * http://msdn.microsoft.com/en-us/library/windows/desktop/aa363911(v=vs.85).aspx
655  */
656  path = f->path;
657  length = wcslen(f->path);
658  if(length >= 9){ /* to ensure that we have at least \??\X:\$x */
659  if(f->path[7] == '$'){
660  volume_letter = (char)f->path[4];
661  for(i = 0; special_file_names[i].original_name; i++){
662  if(winx_wcsistr(f->path,special_file_names[i].original_name)){
663  if(wcslen(f->path) == wcslen(special_file_names[i].original_name) + 0x7){
664  _snwprintf(buffer,MAX_PATH,L"\\??\\%c:\\%ws",volume_letter,
665  special_file_names[i].accepted_name);
666  buffer[MAX_PATH] = 0;
667  path = buffer;
668  DebugPrint("winx_defrag_fopen: %ws used instead of %ws",path,f->path);
669  break;
670  }
671  }
672  }
673  }
674  }
675 
676  RtlInitUnicodeString(&us,path);
677  InitializeObjectAttributes(&oa,&us,0,NULL,NULL);
678  status = NtCreateFile(phandle,access_rights,&oa,&iosb,NULL,0,
679  FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
680  FILE_OPEN,flags,NULL,0);
681  if(status != STATUS_SUCCESS)
682  *phandle = NULL;
683  return status;
684 }
