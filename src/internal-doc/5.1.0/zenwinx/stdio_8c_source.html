328 int winx_prompt_ex(char *prompt,char *string,int n,winx_history *h)
329 {
330  KEYBOARD_INPUT_DATA kbd;
331  KBD_RECORD kbd_rec;
332  char *buffer;
333  int buffer_length;
334  char format[16];
335  int i, ch, line_length;
336  int history_listed_to_the_last_entry = 0;
337 
338  if(!string){
339  winx_printf("\nwinx_prompt_ex: invalid string!\n");
340  return (-1);
341  }
342  if(n <= 0){
343  winx_printf("\nwinx_prompt_ex: invalid string length %d!\n",n);
344  return (-1);
345  }
346 
347  if(!prompt) prompt = "";
348  buffer_length = strlen(prompt) + n;
349  buffer = winx_heap_alloc(buffer_length);
350  if(buffer == NULL){
351  winx_printf("\nNot enough memory for winx_prompt_ex()!\n");
352  return (-1);
353  }
354 
355  winx_printf("%s",prompt);
356 
357  /* keep string always null terminated */
358  RtlZeroMemory(string,n);
359  for(i = 0; i < (n - 1); i ++){
360  /* read keyboard until an ordinary character appearance */
361  do {
362  do{
363  if(kb_read(&kbd,INFINITE) < 0) goto fail;
364  IntTranslateKey(&kbd,&kbd_rec);
365  } while(!kbd_rec.bKeyDown);
366  ch = (int)kbd_rec.AsciiChar;
367  /*
368  * Truncate the string if either backspace or escape pressed.
369  * Walk through history if one of arrow keys pressed.
370  */
371  if(ch == 0x08 || kbd_rec.wVirtualScanCode == 0x1 || \
372  kbd_rec.wVirtualScanCode == 0x48 || kbd_rec.wVirtualScanCode == 0x50){
373  line_length = strlen(prompt) + strlen(string);
374  /* handle escape key */
375  if(kbd_rec.wVirtualScanCode == 0x1){
376  /* truncate the string if escape pressed */
377  RtlZeroMemory(string,n);
378  i = 0;
379  }
380  /* handle backspace key */
381  if(ch == 0x08){
382  /*
383  * make the string one character shorter
384  * if backspace pressed
385  */
386  if(i > 0){
387  i--;
388  string[i] = 0;
389  }
390  }
391  /* handle arrow keys */
392  if(h){
393  if(h->head && h->current){
394  if(kbd_rec.wVirtualScanCode == 0x48){
395  /* list history back */
396  if(h->current == h->head->prev && \
397  !history_listed_to_the_last_entry){
398  /* set the flag and don't list back */
399  history_listed_to_the_last_entry = 1;
400  } else {
401  if(h->current != h->head)
402  h->current = h->current->prev;
403  history_listed_to_the_last_entry = 0;
404  }
405  if(h->current->string){
406  RtlZeroMemory(string,n);
407  strcpy(string,h->current->string);
408  i = strlen(string);
409  }
410  } else if(kbd_rec.wVirtualScanCode == 0x50){
411  /* list history forward */
412  if(h->current->next != h->head){
413  h->current = h->current->next;
414  if(h->current->string){
415  RtlZeroMemory(string,n);
416  strcpy(string,h->current->string);
417  i = strlen(string);
418  }
419  if(h->current == h->head->prev)
420  history_listed_to_the_last_entry = 1;
421  else
422  history_listed_to_the_last_entry = 0;
423  }
424  }
425  }
426  }
427 
428  /* clear history_listed_to_the_last_entry flag */
429  if(ch == 0x08 || kbd_rec.wVirtualScanCode == 0x1)
430  history_listed_to_the_last_entry = 0;
431 
432  /* redraw the prompt */
433  _snprintf(buffer,buffer_length,"%s%s",prompt,string);
434  buffer[buffer_length - 1] = 0;
435  _snprintf(format,sizeof(format),"\r%%-%us",line_length);
436  format[sizeof(format) - 1] = 0;
437  winx_printf(format,buffer);
438  /*
439  * redraw the prompt again to set carriage position
440  * exactly behind the string printed
441  */
442  line_length = strlen(prompt) + strlen(string);
443  _snprintf(format,sizeof(format),"\r%%-%us",line_length);
444  format[sizeof(format) - 1] = 0;
445  winx_printf(format,buffer);
446  continue;
447  }
448  } while(ch == 0 || ch == 0x08 || kbd_rec.wVirtualScanCode == 0x1 || \
449  kbd_rec.wVirtualScanCode == 0x48 || kbd_rec.wVirtualScanCode == 0x50);
450 
451  /* print a character read */
452  winx_putch(ch);
453 
454  /* return when \r character appears */
455  if(ch == '\r'){
456  winx_putch('\n');
457  goto done;
458  }
459 
460  /* we have an ordinary character, append it to the string */
461  string[i] = (char)ch;
462 
463  /* clear the flag in case of ordinary characters typed */
464  history_listed_to_the_last_entry = 0;
465  }
466  winx_printf("\nwinx_prompt_ex: buffer overflow!\n");
467 
468 done:
469  winx_heap_free(buffer);
470  /* add nonempty strings to the history */
471  if(string[0]){
472  winx_add_history_entry(h,string);
473  }
474  return (i+1);
475 
476 fail:
477  winx_heap_free(buffer);
478  return (-1);
479 }

492 static int print_line(char *line_buffer,char *prompt,int max_rows,int *rows_printed,int last_line)
493 {
494  KBD_RECORD kbd_rec;
495  int escape_detected = 0;
496  int break_detected = 0;
497 
498  winx_printf("%s\n",line_buffer);
499  (*rows_printed) ++;
500 
501  if(*rows_printed == max_rows && !last_line){
502  *rows_printed = 0;
503  winx_printf("\n%s\n",prompt);
504  /* wait for any key */
505  if(winx_kb_read(&kbd_rec,INFINITE) < 0){
506  return 1; /* break in case of errors */
507  }
508  /* check for escape */
509  if(kbd_rec.wVirtualScanCode == 0x1){
510  escape_detected = 1;
511  } else if(kbd_rec.wVirtualScanCode == 0x1d){
512  /* distinguish between control keys and break key */
513  if(!(kbd_rec.dwControlKeyState & LEFT_CTRL_PRESSED) && \
514  !(kbd_rec.dwControlKeyState & RIGHT_CTRL_PRESSED)){
515  break_detected = 1;
516  }
517  }
518  winx_printf("\n");
519  if(escape_detected || break_detected) return 1;
520  }
521  return 0;
522 }

552 int winx_print_array_of_strings(char **strings,int line_width,int max_rows,char *prompt,int divide_to_pages)
553 {
554  int i, j, k, index, length;
555  char *line_buffer, *second_buffer;
556  int n, r;
557  int rows_printed;
558 
559  /* check the main parameter for correctness */
560  if(!strings){
561  DebugPrint("winx_print_array_of_strings: strings = NULL!");
562  return (-1);
563  }
564 
565  /* handle situation when text must be displayed entirely */
566  if(!divide_to_pages){
567  for(i = 0; strings[i] != NULL; i++)
568  winx_printf("%s\n",strings[i]);
569  return 0;
570  }
571 
572  /* check other parameters */
573  if(!line_width){
574  DebugPrint("winx_print_array_of_strings: line_width = 0!");
575  return (-1);
576  }
577  if(!max_rows){
578  DebugPrint("winx_print_array_of_strings: max_rows = 0!");
579  return (-1);
580  }
581  if(prompt == NULL) prompt = DEFAULT_PAGING_PROMPT_TO_HIT_ANY_KEY;
582 
583  /* allocate space for prompt on the screen */
584  max_rows -= 4;
585 
586  /* allocate memory for line buffer */
587  line_buffer = winx_heap_alloc(line_width + 1);
588  if(!line_buffer){
589  DebugPrint("winx_print_array_of_strings: cannot allocate %u bytes of memory",
590  line_width + 1);
591  return (-1);
592  }
593  /* allocate memory for second ancillary buffer */
594  second_buffer = winx_heap_alloc(line_width + 1);
595  if(!second_buffer){
596  DebugPrint("winx_print_array_of_strings: cannot allocate %u bytes of memory",
597  line_width + 1);
598  winx_heap_free(line_buffer);
599  return (-1);
600  }
601 
602  /* start to display strings */
603  rows_printed = 0;
604  for(i = 0; strings[i] != NULL; i++){
605  line_buffer[0] = 0;
606  index = 0;
607  length = strlen(strings[i]);
608  for(j = 0; j < length; j++){
609  /* handle \n, \r, \r\n, \n\r sequencies */
610  n = r = 0;
611  if(strings[i][j] == '\n') n = 1;
612  else if(strings[i][j] == '\r') r = 1;
613  if(n || r){
614  /* print buffer */
615  line_buffer[index] = 0;
616  if(print_line(line_buffer,prompt,max_rows,&rows_printed,0))
617  goto cleanup;
618  /* reset buffer */
619  line_buffer[0] = 0;
620  index = 0;
621  /* skip sequence */
622  j++;
623  if(j == length) goto print_rest_of_string;
624  if((strings[i][j] == '\n' && r) || (strings[i][j] == '\r' && n)){
625  continue;
626  } else {
627  if(strings[i][j] == '\n' || strings[i][j] == '\r'){
628  /* process repeating new lines */
629  j--;
630  continue;
631  }
632  /* we have an ordinary character or tabulation -> process them */
633  }
634  }
635  /* handle horizontal tabulation by replacing it by DEFAULT_TAB_WIDTH spaces */
636  if(strings[i][j] == '\t'){
637  for(k = 0; k < DEFAULT_TAB_WIDTH; k++){
638  line_buffer[index] = 0x20;
639  index ++;
640  if(index == line_width){
641  if(j == length - 1) goto print_rest_of_string;
642  line_buffer[index] = 0;
643  if(print_line(line_buffer,prompt,max_rows,&rows_printed,0))
644  goto cleanup;
645  line_buffer[0] = 0;
646  index = 0;
647  break;
648  }
649  }
650  continue;
651  }
652  /* handle ordinary characters */
653  line_buffer[index] = strings[i][j];
654  index ++;
655  if(index == line_width){
656  if(j == length - 1) goto print_rest_of_string;
657  line_buffer[index] = 0;
658  /* break line between words, if possible */
659  for(k = index - 1; k > 0; k--){
660  if(line_buffer[k] == 0x20) break;
661  }
662  if(line_buffer[k] == 0x20){ /* space character found */
663  strcpy(second_buffer,line_buffer + k + 1);
664  line_buffer[k] = 0;
665  if(print_line(line_buffer,prompt,max_rows,&rows_printed,0))
666  goto cleanup;
667  strcpy(line_buffer,second_buffer);
668  index = strlen(line_buffer);
669  } else {
670  if(print_line(line_buffer,prompt,max_rows,&rows_printed,0))
671  goto cleanup;
672  line_buffer[0] = 0;
673  index = 0;
674  }
675  }
676  }
677 print_rest_of_string:
678  line_buffer[index] = 0;
679  if(print_line(line_buffer,prompt,max_rows,&rows_printed,
680  (strings[i+1] == NULL) ? 1 : 0)) goto cleanup;
681  }
682 
683 cleanup:
684  winx_heap_free(line_buffer);
685  winx_heap_free(second_buffer);
686  return 0;
687 }
