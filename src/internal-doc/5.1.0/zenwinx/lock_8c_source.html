43 winx_spin_lock *winx_init_spin_lock(char *name)
44 {
45  winx_spin_lock *sl;
46  char *buffer;
47  int length;
48  wchar_t *fullname;
49 
50  /* attach PID to lock the current process only */
51  buffer = winx_sprintf("\\%s_%u",name,
52  (unsigned int)(DWORD_PTR)(NtCurrentTeb()->ClientId.UniqueProcess));
53  if(buffer == NULL){
54  DebugPrint("winx_init_spin_lock: not enough memory for ansi full name");
55  return NULL;
56  }
57 
58  length = strlen(buffer);
59  fullname = winx_heap_alloc((length + 1) * sizeof(wchar_t));
60  if(fullname == NULL){
61  DebugPrint("winx_init_spin_lock: not enough memory for unicode full name");
62  winx_heap_free(buffer);
63  return NULL;
64  }
65 
66  if(_snwprintf(fullname,length + 1,L"%hs",buffer) < 0){
67  DebugPrint("winx_init_spin_lock: full name conversion to unicode failed");
68  winx_heap_free(buffer);
69  return NULL;
70  }
71 
72  fullname[length] = 0;
73  winx_heap_free(buffer);
74 
75  sl = winx_heap_alloc(sizeof(winx_spin_lock));
76  if(sl == NULL){
77  DebugPrint("winx_init_spin_lock: cannot allocate memory for %s",name);
78  winx_heap_free(fullname);
79  return NULL;
80  }
81 
82  if(winx_create_event(fullname,SynchronizationEvent,&sl->hEvent) < 0){
83  DebugPrint("winx_init_spin_lock: cannot create synchronization event");
84  winx_heap_free(sl);
85  winx_heap_free(fullname);
86  return NULL;
87  }
88 
89  winx_heap_free(fullname);
90 
91  if(winx_release_spin_lock(sl) < 0){
92  winx_destroy_event(sl->hEvent);
93  winx_heap_free(sl);
94  return NULL;
95  }
96 
97  return sl;
98 }

109 int winx_acquire_spin_lock(winx_spin_lock *sl,int msec)
110 {
111  LARGE_INTEGER interval;
112  NTSTATUS status;
113 
114  if(sl == NULL)
115  return (-1);
116 
117  if(sl->hEvent == NULL)
118  return (-1);
119 
120  if(msec != INFINITE)
121  interval.QuadPart = -((signed long)msec * 10000);
122  else
123  interval.QuadPart = MAX_WAIT_INTERVAL;
124  status = NtWaitForSingleObject(sl->hEvent,FALSE,&interval);
125  if(status != WAIT_OBJECT_0)
126  return (-1);
127 
128  return 0;
129 }

136 int winx_release_spin_lock(winx_spin_lock *sl)
